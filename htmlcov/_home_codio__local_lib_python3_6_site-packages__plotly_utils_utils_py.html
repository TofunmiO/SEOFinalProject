28class PlotlyJSONEncoder(_json.JSONEncoder): 

29 """ 

30 Meant to be passed as the `cls` kwarg to json.dumps(obj, cls=..) 

31 

32 See PlotlyJSONEncoder.default for more implementation information. 

33 

34 Additionally, this encoder overrides nan functionality so that 'Inf', 

35 'NaN' and '-Inf' encode to 'null'. Which is stricter JSON than the Python 

36 version. 

37 

38 """

41 """ 

42 This is used to ultimately *encode* into strict JSON, see `encode` 

43 

44 """ 

45 # before python 2.7, 'true', 'false', 'null', were include here. 

46 if const in ("Infinity", "-Infinity", "NaN"): 

47 return None

51 def encode(self, o): 

52 """ 

53 Load and then dump the result using parse_constant kwarg 

54 

55 Note that setting invalid separators will cause a failure at this step. 

56 

57 """ 

58 # this will raise errors in a normal-expected way 

59 encoded_o = super(PlotlyJSONEncoder, self).encode(o) 

60 # Brute force guessing whether NaN or Infinity values are in the string 

61 # We catch false positive cases (e.g. strings such as titles, labels etc.) 

62 # but this is ok since the intention is to skip the decoding / reencoding 

63 # step when it's completely safe 

64 

65 if not ("NaN" in encoded_o or "Infinity" in encoded_o): 

66 return encoded_o 

67 

68 # now: 

69 # 1. `loads` to switch Infinity, -Infinity, NaN to None 

70 # 2. `dumps` again so you get 'null' instead of extended JSON 

71 try: 

72 new_o = _json.loads(encoded_o, parse_constant=self.coerce_to_strict) 

73 except ValueError: 

74 

75 # invalid separators will fail here. raise a helpful exception 

76 raise ValueError( 

77 "Encoding into strict JSON failed. Did you set the separators " 

78 "valid JSON separators?" 

79 ) 

80 else: 

81 return _json.dumps( 

82 new_o, 

83 sort_keys=self.sort_keys, 

84 indent=self.indent, 

85 separators=(self.item_separator, self.key_separator), 

86 )

88 def default(self, obj): 

89 """ 

90 Accept an object (of unknown type) and try to encode with priority: 

91 1. builtin: user-defined objects 

92 2. sage: sage math cloud 

93 3. pandas: dataframes/series 

94 4. numpy: ndarrays 

95 5. datetime: time/datetime objects 

96 

97 Each method throws a NotEncoded exception if it fails. 

98 

99 The default method will only get hit if the object is not a type that 

100 is naturally encoded by json: 

101 

102 Normal objects: 

103 dict object 

104 list, tuple array 

105 str, unicode string 

106 int, long, float number 

107 True true 

108 False false 

109 None null 

110 

111 Extended objects: 

112 float('nan') 'NaN' 

113 float('infinity') 'Infinity' 

114 float('-infinity') '-Infinity' 

115 

116 Therefore, we only anticipate either unknown iterables or values here. 

117 

118 """ 

119 # TODO: The ordering if these methods is *very* important. Is this OK?

122 self.encode_as_sage, 

123 self.encode_as_numpy,

131 for encoding_method in encoding_methods:

332def display_string_positions(p, i=None, offset=0, length=1, char="^", trim=True): 

333 """ 

334 Return a string that is whitespace except at p[i] which is replaced with char. 

335 If i is None then all the indices of the string in p are replaced with char. 

336 

337 Example: 

338 

339 >>> ss = ["a.string[0].with_separators"] 

340 >>> chars = list(".[]_") 

341 >>> ss_split = split_multichar(ss, chars) 

342 >>> ss_split 

343 ['a', 'string', '0', '', 'with', 'separators'] 

344 >>> ss_pos = split_string_positions(ss_split) 

345 >>> ss[0] 

346 'a.string[0].with_separators' 

347 >>> display_string_positions(ss_pos,4) 

348 ' ^' 

349 >>> display_string_positions(ss_pos,4,offset=1,length=3,char="~",trim=False) 

350 ' ~~~ ' 

351 >>> display_string_positions(ss_pos) 

352 '^ ^ ^ ^^ ^' 

353 :param (list) p: A list of integers. 

354 :param (integer|None) i: Optional index of p to display. 

355 :param (integer) offset: Allows adding a number of spaces to the replacement. 

356 :param (integer) length: Allows adding a replacement that is the char 

357 repeated length times. 

358 :param (str) char: allows customizing the replacement character. 

359 :param (boolean) trim: trims the remaining whitespace if True. 

360 """ 

361 s = [" " for _ in range(max(p) + 1 + offset + length)] 

362 maxaddr = 0 

363 if i is None: 

364 for p_ in p: 

365 for l in range(length): 

366 maxaddr = p_ + offset + l 

367 s[maxaddr] = char 

368 else: 

369 for l in range(length): 

370 maxaddr = p[i] + offset + l 

371 s[maxaddr] = char 

372 ret = "".join(s) 

373 if trim: 

374 ret = ret[: maxaddr + 1] 

375 return ret

378def chomp_empty_strings(strings, c, reverse=False): 

379 """ 

380 Given a list of strings, some of which are the empty string "", replace the 

381 empty strings with c and combine them with the closest non-empty string on 

382 the left or "" if it is the first string. 

383 Examples: 

384 for c="_" 

385 ['hey', '', 'why', '', '', 'whoa', '', ''] -> ['hey_', 'why__', 'whoa__'] 

386 ['', 'hi', '', "I'm", 'bob', '', ''] -> ['_', 'hi_', "I'm", 'bob__'] 

387 ['hi', "i'm", 'a', 'good', 'string'] -> ['hi', "i'm", 'a', 'good', 'string'] 

388 Some special cases are: 

389 [] -> [] 

390 [''] -> [''] 

391 ['', ''] -> ['_'] 

392 ['', '', '', ''] -> ['___'] 

393 If reverse is true, empty strings are combined with closest non-empty string 

394 on the right or "" if it is the last string. 

395 """ 

396 

397 def _rev(l): 

398 return [s[::-1] for s in l][::-1] 

399 

400 if reverse: 

401 return _rev(chomp_empty_strings(_rev(strings), c)) 

402 if not len(strings): 

403 return strings 

404 if sum(map(len, strings)) == 0: 

405 return [c * (len(strings) - 1)]

407 class _Chomper: 

408 def __init__(self, c): 

409 self.c = c 

410 

411 def __call__(self, x, y): 

412 # x is list up to now 

413 # y is next item in list 

414 # x should be [""] initially, and then empty strings filtered out at the 

415 # end 

416 if len(y) == 0: 

417 return x[:-1] + [x[-1] + self.c] 

418 else: 

419 return x + [y] 

420 

421 return list(filter(len, reduce(_Chomper(c), strings, [""])))

426def levenshtein(s1, s2): 

427 if len(s1) < len(s2): 

428 return levenshtein(s2, s1) # len(s1) >= len(s2) 

429 if len(s2) == 0: 

430 return len(s1) 

431 previous_row = range(len(s2) + 1) 

432 for i, c1 in enumerate(s1): 

433 current_row = [i + 1] 

434 for j, c2 in enumerate(s2): 

435 # j+1 instead of j since previous_row and current_row are one character longer 

436 # than s2 

437 insertions = previous_row[j + 1] + 1 

438 deletions = current_row[j] + 1 

439 substitutions = previous_row[j] + (c1 != c2) 

440 current_row.append(min(insertions, deletions, substitutions)) 

441 previous_row = current_row 

442 return previous_row[-1]

27class Node: 

28 

29 """A Node is a set of rdatasets.""" 

30 

31 __slots__ = ['rdatasets'] 

32 

33 def __init__(self): 

34 # the set of rdatasets, represented as a list. 

35 self.rdatasets = [] 

36 

37 def to_text(self, name, **kw): 

38 """Convert a node to text format. 

39 

40 Each rdataset at the node is printed. Any keyword arguments 

41 to this method are passed on to the rdataset's to_text() method. 

42 

43 *name*, a ``dns.name.Name`` or ``str``, the owner name of the 

44 rdatasets. 

45 

46 Returns a ``str``. 

47 

48 """ 

49 

50 s = io.StringIO() 

51 for rds in self.rdatasets: 

52 if len(rds) > 0: 

53 s.write(rds.to_text(name, **kw)) 

54 s.write('\n') 

55 return s.getvalue()[:-1]

60 def __eq__(self, other): 

61 # 

62 # This is inefficient. Good thing we don't need to do it much. 

63 # 

64 for rd in self.rdatasets: 

65 if rd not in other.rdatasets: 

66 return False 

67 for rd in other.rdatasets: 

68 if rd not in self.rdatasets: 

69 return False 

70 return True

81 def find_rdataset(self, rdclass, rdtype, covers=dns.rdatatype.NONE, 

82 create=False): 

83 """Find an rdataset matching the specified properties in the 

84 current node. 

85 

86 *rdclass*, an ``int``, the class of the rdataset. 

87 

88 *rdtype*, an ``int``, the type of the rdataset. 

89 

90 *covers*, an ``int`` or ``None``, the covered type. 

91 Usually this value is ``dns.rdatatype.NONE``, but if the 

92 rdtype is ``dns.rdatatype.SIG`` or ``dns.rdatatype.RRSIG``, 

93 then the covers value will be the rdata type the SIG/RRSIG 

94 covers. The library treats the SIG and RRSIG types as if they 

95 were a family of types, e.g. RRSIG(A), RRSIG(NS), RRSIG(SOA). 

96 This makes RRSIGs much easier to work with than if RRSIGs 

97 covering different rdata types were aggregated into a single 

98 RRSIG rdataset. 

99 

100 *create*, a ``bool``. If True, create the rdataset if it is not found. 

101 

102 Raises ``KeyError`` if an rdataset of the desired type and class does 

103 not exist and *create* is not ``True``. 

104 

105 Returns a ``dns.rdataset.Rdataset``. 

106 """ 

107 

108 for rds in self.rdatasets: 

109 if rds.match(rdclass, rdtype, covers): 

110 return rds 

111 if not create: 

112 raise KeyError 

113 rds = dns.rdataset.Rdataset(rdclass, rdtype) 

114 self.rdatasets.append(rds) 

115 return rds

117 def get_rdataset(self, rdclass, rdtype, covers=dns.rdatatype.NONE, 

118 create=False): 

119 """Get an rdataset matching the specified properties in the 

120 current node. 

121 

122 None is returned if an rdataset of the specified type and 

123 class does not exist and *create* is not ``True``. 

124 

125 *rdclass*, an ``int``, the class of the rdataset. 

126 

127 *rdtype*, an ``int``, the type of the rdataset. 

128 

129 *covers*, an ``int``, the covered type. Usually this value is 

130 dns.rdatatype.NONE, but if the rdtype is dns.rdatatype.SIG or 

131 dns.rdatatype.RRSIG, then the covers value will be the rdata 

132 type the SIG/RRSIG covers. The library treats the SIG and RRSIG 

133 types as if they were a family of 

134 types, e.g. RRSIG(A), RRSIG(NS), RRSIG(SOA). This makes RRSIGs much 

135 easier to work with than if RRSIGs covering different rdata 

136 types were aggregated into a single RRSIG rdataset. 

137 

138 *create*, a ``bool``. If True, create the rdataset if it is not found. 

139 

140 Returns a ``dns.rdataset.Rdataset`` or ``None``. 

141 """ 

142 

143 try: 

144 rds = self.find_rdataset(rdclass, rdtype, covers, create) 

145 except KeyError: 

146 rds = None 

147 return rds 

148 

149 def delete_rdataset(self, rdclass, rdtype, covers=dns.rdatatype.NONE): 

150 """Delete the rdataset matching the specified properties in the 

151 current node. 

152 

153 If a matching rdataset does not exist, it is not an error. 

154 

155 *rdclass*, an ``int``, the class of the rdataset. 

156 

157 *rdtype*, an ``int``, the type of the rdataset. 

158 

159 *covers*, an ``int``, the covered type. 

160 """ 

161 

162 rds = self.get_rdataset(rdclass, rdtype, covers) 

163 if rds is not None: 

164 self.rdatasets.remove(rds) 

165 

166 def replace_rdataset(self, replacement): 

167 """Replace an rdataset. 

168 

169 It is not an error if there is no rdataset matching *replacement*. 

170 

171 Ownership of the *replacement* object is transferred to the node; 

172 in other words, this method does not store a copy of *replacement* 

173 at the node, it stores *replacement* itself. 

174 

175 *replacement*, a ``dns.rdataset.Rdataset``. 

176 

177 Raises ``ValueError`` if *replacement* is not a 

178 ``dns.rdataset.Rdataset``. 

179 """ 

180 

181 if not isinstance(replacement, dns.rdataset.Rdataset): 

182 raise ValueError('replacement is not an rdataset') 

183 if isinstance(replacement, dns.rrset.RRset): 

184 # RRsets are not good replacements as the match() method 

185 # is not compatible. 

186 replacement = replacement.to_rdataset() 

187 self.delete_rdataset(replacement.rdclass, replacement.rdtype, 

188 replacement.covers) 

189 self.rdatasets.append(replacement)

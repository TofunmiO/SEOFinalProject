30 def origin_information(self): 

31 """Returns a tuple 

32 

33 (absolute_origin, relativize, effective_origin)

36 relative domain names, the "effective origin", and whether 

37 names should be relativized. The "effective origin" is the 

38 absolute origin if relativize is False, and the empty name if

94 def get(self, name, rdtype, covers=dns.rdatatype.NONE): 

95 """Return the rdataset associated with *name*, *rdtype*, and *covers*, 

96 or `None` if not found. 

97 

98 Note that the returned rdataset is immutable. 

99 """ 

100 self._check_ended() 

101 if isinstance(name, str): 

102 name = dns.name.from_text(name, None) 

103 rdtype = dns.rdatatype.RdataType.make(rdtype) 

104 rdataset = self._get_rdataset(name, rdtype, covers) 

105 if rdataset is not None and \ 

106 not isinstance(rdataset, dns.rdataset.ImmutableRdataset): 

107 rdataset = dns.rdataset.ImmutableRdataset(rdataset) 

108 return rdataset

202 def update_serial(self, value=1, relative=True, name=dns.name.empty): 

203 """Update the serial number. 

204 

205 *value*, an `int`, is an increment if *relative* is `True`, or the 

206 actual value to set if *relative* is `False`. 

207 

208 Raises `KeyError` if there is no SOA rdataset at *name*. 

209 

210 Raises `ValueError` if *value* is negative or if the increment is 

211 so large that it would cause the new serial to be less than the 

212 prior value. 

213 """ 

214 self._check_ended() 

215 if value < 0: 

216 raise ValueError('negative update_serial() value') 

217 if isinstance(name, str): 

218 name = dns.name.from_text(name, None) 

219 rdataset = self._get_rdataset(name, dns.rdatatype.SOA, 

220 dns.rdatatype.NONE) 

221 if rdataset is None or len(rdataset) == 0: 

222 raise KeyError 

223 if relative: 

224 serial = dns.serial.Serial(rdataset[0].serial) + value 

225 else: 

226 serial = dns.serial.Serial(value) 

227 serial = serial.value # convert back to int 

228 if serial == 0: 

229 serial = 1 

230 rdata = rdataset[0].replace(serial=serial) 

231 new_rdataset = dns.rdataset.from_rdata(rdataset.ttl, rdata) 

232 self.replace(name, new_rdataset)

249 def commit(self): 

250 """Commit the transaction. 

251 

252 Normally transactions are used as context managers and commit

254 A ``dns.transaction.Ended`` exception will be raised if you try 

255 to use a transaction after it has been committed or rolled back.

257 Raises an exception if the commit fails (in which case the transaction 

258 is also rolled back. 

259 """ 

260 self._end(True)

312 def _add(self, replace, args): 

313 try: 

314 args = collections.deque(args) 

315 if replace: 

316 method = 'replace()' 

317 else: 

318 method = 'add()' 

319 arg = args.popleft() 

320 if isinstance(arg, str): 

321 arg = dns.name.from_text(arg, None) 

322 if isinstance(arg, dns.name.Name): 

323 name = arg 

324 rdataset = self._rdataset_from_args(method, False, args) 

325 elif isinstance(arg, dns.rrset.RRset): 

326 rrset = arg 

327 name = rrset.name 

328 # rrsets are also rdatasets, but they don't print the 

329 # same and can't be stored in nodes, so convert. 

330 rdataset = rrset.to_rdataset() 

331 else: 

332 raise TypeError(f'{method} requires a name or RRset ' + 

333 'as the first argument') 

334 if rdataset.rdclass != self.manager.get_class(): 

335 raise ValueError(f'{method} has objects of wrong RdataClass') 

336 if rdataset.rdtype == dns.rdatatype.SOA: 

337 (_, _, origin) = self.manager.origin_information() 

338 if name != origin: 

339 raise ValueError(f'{method} has non-origin SOA') 

340 self._raise_if_not_empty(method, args) 

341 if not replace: 

342 existing = self._get_rdataset(name, rdataset.rdtype, 

343 rdataset.covers) 

344 if existing is not None: 

345 if isinstance(existing, dns.rdataset.ImmutableRdataset): 

346 trds = dns.rdataset.Rdataset(existing.rdclass, 

347 existing.rdtype, 

348 existing.covers) 

349 trds.update(existing) 

350 existing = trds 

351 rdataset = existing.union(rdataset) 

352 self._put_rdataset(name, rdataset) 

353 except IndexError: 

354 raise TypeError(f'not enough parameters to {method}')

356 def _delete(self, exact, args): 

357 try: 

358 args = collections.deque(args) 

359 if exact: 

360 method = 'delete_exact()' 

361 else: 

362 method = 'delete()' 

363 arg = args.popleft() 

364 if isinstance(arg, str): 

365 arg = dns.name.from_text(arg, None) 

366 if isinstance(arg, dns.name.Name): 

367 name = arg 

368 if len(args) > 0 and (isinstance(args[0], int) or 

369 isinstance(args[0], str)): 

370 # deleting by type and (optionally) covers 

371 rdtype = dns.rdatatype.RdataType.make(args.popleft()) 

372 if len(args) > 0: 

373 covers = dns.rdatatype.RdataType.make(args.popleft()) 

374 else: 

375 covers = dns.rdatatype.NONE 

376 self._raise_if_not_empty(method, args) 

377 existing = self._get_rdataset(name, rdtype, covers) 

378 if existing is None: 

379 if exact: 

380 raise DeleteNotExact(f'{method}: missing rdataset') 

381 else: 

382 self._delete_rdataset(name, rdtype, covers) 

383 return 

384 else: 

385 rdataset = self._rdataset_from_args(method, True, args) 

386 elif isinstance(arg, dns.rrset.RRset): 

387 rdataset = arg # rrsets are also rdatasets 

388 name = rdataset.name 

389 else: 

390 raise TypeError(f'{method} requires a name or RRset ' + 

391 'as the first argument') 

392 self._raise_if_not_empty(method, args) 

393 if rdataset: 

394 if rdataset.rdclass != self.manager.get_class(): 

395 raise ValueError(f'{method} has objects of wrong ' 

396 'RdataClass') 

397 existing = self._get_rdataset(name, rdataset.rdtype, 

398 rdataset.covers) 

399 if existing is not None: 

400 if exact: 

401 intersection = existing.intersection(rdataset) 

402 if intersection != rdataset: 

403 raise DeleteNotExact(f'{method}: missing rdatas') 

404 rdataset = existing.difference(rdataset) 

405 if len(rdataset) == 0: 

406 self._delete_rdataset(name, rdataset.rdtype, 

407 rdataset.covers) 

408 else: 

409 self._put_rdataset(name, rdataset) 

410 elif exact: 

411 raise DeleteNotExact(f'{method}: missing rdataset') 

412 else: 

413 if exact and not self._name_exists(name): 

414 raise DeleteNotExact(f'{method}: name not known') 

415 self._delete_name(name) 

416 except IndexError: 

417 raise TypeError(f'not enough parameters to {method}')

419 def _check_ended(self): 

420 if self._ended: 

421 raise AlreadyEnded

423 def _end(self, commit): 

424 self._check_ended() 

425 if self._ended: 

426 raise AlreadyEnded 

427 try: 

428 self._end_transaction(commit) 

429 finally: 

430 self._ended = True

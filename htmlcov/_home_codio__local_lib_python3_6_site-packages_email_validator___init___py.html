11# Based on RFC 2822 section 3.2.4 / RFC 5322 section 3.2.3, these 

12# characters are permitted in email addresses (not taking into 

13# account internationalization): 

14ATEXT = r'a-zA-Z0-9_!#\$%&\'\*\+\-/=\?\^`\{\|\}~' 

15 

16# A "dot atom text", per RFC 2822 3.2.4: 

17DOT_ATOM_TEXT = '[' + ATEXT + ']+(?:\\.[' + ATEXT + ']+)*' 

18 

19# RFC 6531 section 3.3 extends the allowed characters in internationalized 

20# addresses to also include three specific ranges of UTF8 defined in 

21# RFC3629 section 4, which appear to be the Unicode code points from 

22# U+0080 to U+10FFFF. 

23ATEXT_UTF8 = ATEXT + u"\u0080-\U0010FFFF" 

24DOT_ATOM_TEXT_UTF8 = '[' + ATEXT_UTF8 + ']+(?:\\.[' + ATEXT_UTF8 + ']+)*' 

25 

26# The domain part of the email address, after IDNA (ASCII) encoding, 

27# must also satisfy the requirements of RFC 952/RFC 1123 which restrict 

28# the allowed characters of hostnames further. The hyphen cannot be at 

29# the beginning or end of a *dot-atom component* of a hostname either. 

30ATEXT_HOSTNAME = r'(?:(?:[a-zA-Z0-9][a-zA-Z0-9\-]*)?[a-zA-Z0-9])'

190def validate_email( 

191 email, 

192 allow_smtputf8=True, 

193 allow_empty_local=False, 

194 check_deliverability=True, 

195 timeout=DEFAULT_TIMEOUT, 

196 dns_resolver=None 

197): 

198 """ 

199 Validates an email address, raising an EmailNotValidError if the address is not valid or returning a dict of 

200 information when the address is valid. The email argument can be a str or a bytes instance, 

201 but if bytes it must be ASCII-only. 

202 """ 

203 

204 # Allow email to be a str or bytes instance. If bytes, 

205 # it must be ASCII because that's how the bytes work 

206 # on the wire with SMTP. 

207 if not isinstance(email, (str, unicode_class)): 

208 try: 

209 email = email.decode("ascii") 

210 except ValueError: 

211 raise EmailSyntaxError("The email address is not valid ASCII.") 

212 

213 # At-sign. 

214 parts = email.split('@') 

215 if len(parts) != 2: 

216 raise EmailSyntaxError("The email address is not valid. It must have exactly one @-sign.") 

217 

218 # Collect return values in this instance. 

219 ret = ValidatedEmail() 

220 ret.original_email = email 

221 

222 # Validate the email address's local part syntax and get a normalized form. 

223 local_part_info = validate_email_local_part(parts[0], 

224 allow_smtputf8=allow_smtputf8, 

225 allow_empty_local=allow_empty_local) 

226 ret.local_part = local_part_info["local_part"] 

227 ret.ascii_local_part = local_part_info["ascii_local_part"] 

228 ret.smtputf8 = local_part_info["smtputf8"] 

229 

230 # Validate the email address's domain part syntax and get a normalized form. 

231 domain_part_info = validate_email_domain_part(parts[1]) 

232 ret.domain = domain_part_info["domain"] 

233 ret.ascii_domain = domain_part_info["ascii_domain"] 

234 

235 # Construct the complete normalized form. 

236 ret.email = ret.local_part + "@" + ret.domain 

237 

238 # If the email address has an ASCII form, add it. 

239 if not ret.smtputf8: 

240 ret.ascii_email = ret.ascii_local_part + "@" + ret.ascii_domain 

241 

242 # If the email address has an ASCII representation, then we assume it may be 

243 # transmitted in ASCII (we can't assume SMTPUTF8 will be used on all hops to 

244 # the destination) and the length limit applies to ASCII characters (which is 

245 # the same as octets). The number of characters in the internationalized form 

246 # may be many fewer (because IDNA ASCII is verbose) and could be less than 254 

247 # Unicode characters, and of course the number of octets over the limit may 

248 # not be the number of characters over the limit, so if the email address is 

249 # internationalized, we can't give any simple information about why the address 

250 # is too long. 

251 # 

252 # In addition, check that the UTF-8 encoding (i.e. not IDNA ASCII and not 

253 # Unicode characters) is at most 254 octets. If the addres is transmitted using 

254 # SMTPUTF8, then the length limit probably applies to the UTF-8 encoded octets. 

255 # If the email address has an ASCII form that differs from its internationalized 

256 # form, I don't think the internationalized form can be longer, and so the ASCII 

257 # form length check would be sufficient. If there is no ASCII form, then we have 

258 # to check the UTF-8 encoding. The UTF-8 encoding could be up to about four times 

259 # longer than the number of characters. 

260 # 

261 # See the length checks on the local part and the domain. 

262 if ret.ascii_email and len(ret.ascii_email) > EMAIL_MAX_LENGTH: 

263 if ret.ascii_email == ret.email: 

264 reason = __get_length_reason(ret.ascii_email) 

265 elif len(ret.email) > EMAIL_MAX_LENGTH: 

266 # If there are more than 254 characters, then the ASCII 

267 # form is definitely going to be too long. 

268 reason = __get_length_reason(ret.email, utf8=True) 

269 else: 

270 reason = "(when converted to IDNA ASCII)" 

271 raise EmailSyntaxError("The email address is too long {}.".format(reason)) 

272 if len(ret.email.encode("utf8")) > EMAIL_MAX_LENGTH: 

273 if len(ret.email) > EMAIL_MAX_LENGTH: 

274 # If there are more than 254 characters, then the UTF-8 

275 # encoding is definitely going to be too long. 

276 reason = __get_length_reason(ret.email, utf8=True) 

277 else: 

278 reason = "(when encoded in bytes)" 

279 raise EmailSyntaxError("The email address is too long {}.".format(reason)) 

280 

281 if check_deliverability: 

282 # Validate the email address's deliverability and update the 

283 # return dict with metadata. 

284 deliverability_info = validate_email_deliverability( 

285 ret["domain"], ret["domain_i18n"], timeout, dns_resolver 

286 ) 

287 if "mx" in deliverability_info: 

288 ret.mx = deliverability_info["mx"] 

289 ret.mx_fallback_type = deliverability_info["mx-fallback"] 

290 

291 return ret

294def validate_email_local_part(local, allow_smtputf8=True, allow_empty_local=False): 

295 # Validates the local part of an email address. 

296 

297 if len(local) == 0: 

298 if not allow_empty_local: 

299 raise EmailSyntaxError("There must be something before the @-sign.") 

300 else: 

301 # The caller allows an empty local part. Useful for validating certain 

302 # Postfix aliases. 

303 return { 

304 "local_part": local, 

305 "ascii_local_part": local, 

306 "smtputf8": False, 

307 } 

308 

309 # RFC 5321 4.5.3.1.1 

310 # We're checking the number of characters here. If the local part 

311 # is ASCII-only, then that's the same as bytes (octets). If it's 

312 # internationalized, then the UTF-8 encoding may be longer, but 

313 # that may not be relevant. We will check the total address length 

314 # instead. 

315 if len(local) > LOCAL_PART_MAX_LENGTH: 

316 reason = __get_length_reason(local, limit=LOCAL_PART_MAX_LENGTH) 

317 raise EmailSyntaxError("The email address is too long before the @-sign {}.".format(reason)) 

318 

319 # Check the local part against the regular expression for the older ASCII requirements. 

320 m = re.match(DOT_ATOM_TEXT + "\\Z", local) 

321 if m: 

322 # Return the local part unchanged and flag that SMTPUTF8 is not needed. 

323 return { 

324 "local_part": local, 

325 "ascii_local_part": local, 

326 "smtputf8": False, 

327 } 

328 

329 else: 

330 # The local part failed the ASCII check. Now try the extended internationalized requirements. 

331 m = re.match(DOT_ATOM_TEXT_UTF8 + "\\Z", local) 

332 if not m: 

333 # It's not a valid internationalized address either. Report which characters were not valid. 

334 bad_chars = ', '.join(sorted(set( 

335 c for c in local if not re.match(u"[" + (ATEXT if not allow_smtputf8 else ATEXT_UTF8) + u"]", c) 

336 ))) 

337 raise EmailSyntaxError("The email address contains invalid characters before the @-sign: %s." % bad_chars) 

338 

339 # It would be valid if internationalized characters were allowed by the caller. 

340 if not allow_smtputf8: 

341 raise EmailSyntaxError("Internationalized characters before the @-sign are not supported.") 

342 

343 # It's valid. 

344 

345 # RFC 6532 section 3.1 also says that Unicode NFC normalization should be applied, 

346 # so we'll return the normalized local part in the return value. 

347 local = unicodedata.normalize("NFC", local) 

348 

349 # Flag that SMTPUTF8 will be required for deliverability. 

350 return { 

351 "local_part": local, 

352 "ascii_local_part": None, # no ASCII form is possible 

353 "smtputf8": True, 

354 }

357def validate_email_domain_part(domain): 

358 # Empty? 

359 if len(domain) == 0: 

360 raise EmailSyntaxError("There must be something after the @-sign.") 

361 

362 # Perform UTS-46 normalization, which includes casefolding, NFC normalization, 

363 # and converting all label separators (the period/full stop, fullwidth full stop, 

364 # ideographic full stop, and halfwidth ideographic full stop) to basic periods. 

365 # It will also raise an exception if there is an invalid character in the input, 

366 # such as "⒈" which is invalid because it would expand to include a period. 

367 try: 

368 domain = idna.uts46_remap(domain, std3_rules=False, transitional=False) 

369 except idna.IDNAError as e: 

370 raise EmailSyntaxError("The domain name %s contains invalid characters (%s)." % (domain, str(e))) 

371 

372 # Now we can perform basic checks on the use of periods (since equivalent 

373 # symbols have been mapped to periods). These checks are needed because the 

374 # IDNA library doesn't handle well domains that have empty labels (i.e. initial 

375 # dot, trailing dot, or two dots in a row). 

376 if domain.endswith("."): 

377 raise EmailSyntaxError("An email address cannot end with a period.") 

378 if domain.startswith("."): 

379 raise EmailSyntaxError("An email address cannot have a period immediately after the @-sign.") 

380 if ".." in domain: 

381 raise EmailSyntaxError("An email address cannot have two periods in a row.") 

382 

383 # Regardless of whether international characters are actually used, 

384 # first convert to IDNA ASCII. For ASCII-only domains, the transformation 

385 # does nothing. If internationalized characters are present, the MTA 

386 # must either support SMTPUTF8 or the mail client must convert the 

387 # domain name to IDNA before submission. 

388 # 

389 # Unfortunately this step incorrectly 'fixes' domain names with leading 

390 # periods by removing them, so we have to check for this above. It also gives 

391 # a funky error message ("No input") when there are two periods in a 

392 # row, also checked separately above. 

393 try: 

394 ascii_domain = idna.encode(domain, uts46=False).decode("ascii") 

395 except idna.IDNAError as e: 

396 if "Domain too long" in str(e): 

397 # We can't really be more specific because UTS-46 normalization means 

398 # the length check is applied to a string that is different from the 

399 # one the user supplied. Also I'm not sure if the length check applies 

400 # to the internationalized form, the IDNA ASCII form, or even both! 

401 raise EmailSyntaxError("The email address is too long after the @-sign.") 

402 raise EmailSyntaxError("The domain name %s contains invalid characters (%s)." % (domain, str(e))) 

403 

404 # We may have been given an IDNA ASCII domain to begin with. Check 

405 # that the domain actually conforms to IDNA. It could look like IDNA 

406 # but not be actual IDNA. For ASCII-only domains, the conversion out 

407 # of IDNA just gives the same thing back. 

408 # 

409 # This gives us the canonical internationalized form of the domain, 

410 # which we should use in all error messages. 

411 try: 

412 domain_i18n = idna.decode(ascii_domain.encode('ascii')) 

413 except idna.IDNAError as e: 

414 raise EmailSyntaxError("The domain name %s is not valid IDNA (%s)." % (ascii_domain, str(e))) 

415 

416 # RFC 5321 4.5.3.1.2 

417 # We're checking the number of bytes (octets) here, which can be much 

418 # higher than the number of characters in internationalized domains, 

419 # on the assumption that the domain may be transmitted without SMTPUTF8 

420 # as IDNA ASCII. This is also checked by idna.encode, so this exception 

421 # is never reached. 

422 if len(ascii_domain) > DOMAIN_MAX_LENGTH: 

423 raise EmailSyntaxError("The email address is too long after the @-sign.") 

424 

425 # A "dot atom text", per RFC 2822 3.2.4, but using the restricted 

426 # characters allowed in a hostname (see ATEXT_HOSTNAME above). 

427 DOT_ATOM_TEXT = ATEXT_HOSTNAME + r'(?:\.' + ATEXT_HOSTNAME + r')*' 

428 

429 # Check the regular expression. This is probably entirely redundant 

430 # with idna.decode, which also checks this format. 

431 m = re.match(DOT_ATOM_TEXT + "\\Z", ascii_domain) 

432 if not m: 

433 raise EmailSyntaxError("The email address contains invalid characters after the @-sign.") 

434 

435 # All publicly deliverable addresses have domain named with at least 

436 # one period. We also know that all TLDs end with a letter. 

437 if "." not in ascii_domain: 

438 raise EmailSyntaxError("The domain name %s is not valid. It should have a period." % domain_i18n) 

439 if not re.search(r"[A-Za-z]\Z", ascii_domain): 

440 raise EmailSyntaxError( 

441 "The domain name %s is not valid. It is not within a valid top-level domain." % domain_i18n 

442 ) 

443 

444 # Return the IDNA ASCII-encoded form of the domain, which is how it 

445 # would be transmitted on the wire (except when used with SMTPUTF8 

446 # possibly), as well as the canonical Unicode form of the domain, 

447 # which is better for display purposes. This should also take care 

448 # of RFC 6532 section 3.1's suggestion to apply Unicode NFC 

449 # normalization to addresses. 

450 return { 

451 "ascii_domain": ascii_domain, 

452 "domain": domain_i18n, 

453 }

456def validate_email_deliverability(domain, domain_i18n, timeout=DEFAULT_TIMEOUT, dns_resolver=None): 

457 # Check that the domain resolves to an MX record. If there is no MX record, 

458 # try an A or AAAA record which is a deprecated fallback for deliverability. 

459 

460 # If no dns.resolver.Resolver was given, get dnspython's default resolver. 

461 # Override the default resolver's timeout. This may affect other uses of 

462 # dnspython in this process. 

463 if dns_resolver is None: 

464 dns_resolver = dns.resolver.get_default_resolver() 

465 dns_resolver.lifetime = timeout 

466 

467 def dns_resolver_resolve_shim(domain, record): 

468 try: 

469 # dns.resolver.Resolver.resolve is new to dnspython 2.x. 

470 # https://dnspython.readthedocs.io/en/latest/resolver-class.html#dns.resolver.Resolver.resolve 

471 return dns_resolver.resolve(domain, record) 

472 except AttributeError: 

473 # dnspython 2.x is only available in Python 3.6 and later. For earlier versions 

474 # of Python, we maintain compatibility with dnspython 1.x which has a 

475 # dnspython.resolver.Resolver.query method instead. The only difference is that 

476 # query may treat the domain as relative and use the system's search domains, 

477 # which we prevent by adding a "." to the domain name to make it absolute. 

478 # dns.resolver.Resolver.query is deprecated in dnspython version 2.x. 

479 # https://dnspython.readthedocs.io/en/latest/resolver-class.html#dns.resolver.Resolver.query 

480 return dns_resolver.query(domain + ".", record) 

481 

482 try: 

483 # We need a way to check how timeouts are handled in the tests. So we 

484 # have a secret variable that if set makes this method always test the 

485 # handling of a timeout. 

486 if getattr(validate_email_deliverability, 'TEST_CHECK_TIMEOUT', False): 

487 raise dns.exception.Timeout() 

488 

489 try: 

490 # Try resolving for MX records and get them in sorted priority order. 

491 response = dns_resolver_resolve_shim(domain, "MX") 

492 mtas = sorted([(r.preference, str(r.exchange).rstrip('.')) for r in response]) 

493 mx_fallback = None 

494 except (dns.resolver.NoNameservers, dns.resolver.NXDOMAIN, dns.resolver.NoAnswer): 

495 

496 # If there was no MX record, fall back to an A record. 

497 try: 

498 response = dns_resolver_resolve_shim(domain, "A") 

499 mtas = [(0, str(r)) for r in response] 

500 mx_fallback = "A" 

501 except (dns.resolver.NoNameservers, dns.resolver.NXDOMAIN, dns.resolver.NoAnswer): 

502 

503 # If there was no A record, fall back to an AAAA record. 

504 try: 

505 response = dns_resolver_resolve_shim(domain, "AAAA") 

506 mtas = [(0, str(r)) for r in response] 

507 mx_fallback = "AAAA" 

508 except (dns.resolver.NoNameservers, dns.resolver.NXDOMAIN, dns.resolver.NoAnswer): 

509 

510 # If there was no MX, A, or AAAA record, then mail to 

511 # this domain is not deliverable. 

512 raise EmailUndeliverableError("The domain name %s does not exist." % domain_i18n) 

513 

514 except dns.exception.Timeout: 

515 # A timeout could occur for various reasons, so don't treat it as a failure. 

516 return { 

517 "unknown-deliverability": "timeout", 

518 } 

519 

520 except EmailUndeliverableError: 

521 # Don't let these get clobbered by the wider except block below. 

522 raise 

523 

524 except Exception as e: 

525 # Unhandled conditions should not propagate. 

526 raise EmailUndeliverableError( 

527 "There was an error while checking if the domain name in the email address is deliverable: " + str(e) 

528 ) 

529 

530 return { 

531 "mx": mtas, 

532 "mx-fallback": mx_fallback, 

533 }

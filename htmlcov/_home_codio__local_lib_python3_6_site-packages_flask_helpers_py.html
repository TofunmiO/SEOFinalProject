66def stream_with_context( 

67 generator_or_function: t.Union[ 

68 t.Iterator[t.AnyStr], t.Callable[..., t.Iterator[t.AnyStr]] 

69 ] 

70) -> t.Iterator[t.AnyStr]: 

71 """Request contexts disappear when the response is started on the server. 

72 This is done for efficiency reasons and to make it less likely to encounter 

73 memory leaks with badly written WSGI middlewares. The downside is that if 

74 you are using streamed responses, the generator cannot access request bound 

75 information any more. 

76 

77 This function however can help you keep the context around for longer:: 

78 

79 from flask import stream_with_context, request, Response 

80 

81 @app.route('/stream') 

82 def streamed_response(): 

83 @stream_with_context 

84 def generate(): 

85 yield 'Hello ' 

86 yield request.args['name'] 

87 yield '!' 

88 return Response(generate()) 

89 

90 Alternatively it can also be used around a specific generator:: 

91 

92 from flask import stream_with_context, request, Response 

93 

94 @app.route('/stream') 

95 def streamed_response(): 

96 def generate(): 

97 yield 'Hello ' 

98 yield request.args['name'] 

99 yield '!' 

100 return Response(stream_with_context(generate())) 

101 

102 .. versionadded:: 0.9 

103 """ 

104 try: 

105 gen = iter(generator_or_function) # type: ignore 

106 except TypeError: 

107 

108 def decorator(*args: t.Any, **kwargs: t.Any) -> t.Any: 

109 gen = generator_or_function(*args, **kwargs) # type: ignore 

110 return stream_with_context(gen) 

111 

112 return update_wrapper(decorator, generator_or_function) # type: ignore 

113 

114 def generator() -> t.Generator: 

115 ctx = _request_ctx_stack.top 

116 if ctx is None: 

117 raise RuntimeError( 

118 "Attempted to stream with context but " 

119 "there was no context in the first place to keep around." 

120 ) 

121 with ctx: 

122 # Dummy sentinel. Has to be inside the context block or we're 

123 # not actually keeping the context around. 

124 yield None 

125 

126 # The try/finally is here so that if someone passes a WSGI level 

127 # iterator in we're still running the cleanup logic. Generators 

128 # don't need that because they are closed on their destruction 

129 # automatically. 

130 try: 

131 yield from gen 

132 finally: 

133 if hasattr(gen, "close"): 

134 gen.close() # type: ignore 

135 

136 # The trick is to start the generator. Then the code execution runs until 

137 # the first dummy None is yielded at which point the context was already 

138 # pushed. This item is discarded. Then when the iteration continues the 

139 # real generator is executed. 

140 wrapped_g = generator() 

141 next(wrapped_g) 

142 return wrapped_g

145def make_response(*args: t.Any) -> "Response": 

146 """Sometimes it is necessary to set additional headers in a view. Because 

147 views do not have to return response objects but can return a value that 

148 is converted into a response object by Flask itself, it becomes tricky to 

149 add headers to it. This function can be called instead of using a return 

150 and you will get a response object which you can use to attach headers. 

151 

152 If view looked like this and you want to add a new header:: 

153 

154 def index(): 

155 return render_template('index.html', foo=42) 

156 

157 You can now do something like this:: 

158 

159 def index(): 

160 response = make_response(render_template('index.html', foo=42)) 

161 response.headers['X-Parachutes'] = 'parachutes are cool' 

162 return response 

163 

164 This function accepts the very same arguments you can return from a 

165 view function. This for example creates a response with a 404 error 

166 code:: 

167 

168 response = make_response(render_template('not_found.html'), 404) 

169 

170 The other use case of this function is to force the return value of a 

171 view function into a response which is helpful with view 

172 decorators:: 

173 

174 response = make_response(view_function()) 

175 response.headers['X-Parachutes'] = 'parachutes are cool' 

176 

177 Internally this function does the following things: 

178 

179 - if no arguments are passed, it creates a new response argument 

180 - if one argument is passed, :meth:`flask.Flask.make_response` 

181 is invoked with it. 

182 - if more than one argument is passed, the arguments are passed 

183 to the :meth:`flask.Flask.make_response` function as tuple. 

184 

185 .. versionadded:: 0.6 

186 """ 

187 if not args: 

188 return current_app.response_class() 

189 if len(args) == 1: 

190 args = args[0] 

191 return current_app.make_response(args)

194def url_for(endpoint: str, **values: t.Any) -> str: 

195 """Generates a URL to the given endpoint with the method provided. 

196 

197 Variable arguments that are unknown to the target endpoint are appended 

198 to the generated URL as query arguments. If the value of a query argument 

199 is ``None``, the whole pair is skipped. In case blueprints are active 

200 you can shortcut references to the same blueprint by prefixing the 

201 local endpoint with a dot (``.``). 

202 

203 This will reference the index function local to the current blueprint:: 

204 

205 url_for('.index') 

206 

207 See :ref:`url-building`. 

208 

209 Configuration values ``APPLICATION_ROOT`` and ``SERVER_NAME`` are only used when 

210 generating URLs outside of a request context. 

211 

212 To integrate applications, :class:`Flask` has a hook to intercept URL build 

213 errors through :attr:`Flask.url_build_error_handlers`. The `url_for` 

214 function results in a :exc:`~werkzeug.routing.BuildError` when the current 

215 app does not have a URL for the given endpoint and values. When it does, the 

216 :data:`~flask.current_app` calls its :attr:`~Flask.url_build_error_handlers` if 

217 it is not ``None``, which can return a string to use as the result of 

218 `url_for` (instead of `url_for`'s default to raise the 

219 :exc:`~werkzeug.routing.BuildError` exception) or re-raise the exception. 

220 An example:: 

221 

222 def external_url_handler(error, endpoint, values): 

223 "Looks up an external URL when `url_for` cannot build a URL." 

224 # This is an example of hooking the build_error_handler. 

225 # Here, lookup_url is some utility function you've built 

226 # which looks up the endpoint in some external URL registry. 

227 url = lookup_url(endpoint, **values) 

228 if url is None: 

229 # External lookup did not have a URL. 

230 # Re-raise the BuildError, in context of original traceback. 

231 exc_type, exc_value, tb = sys.exc_info() 

232 if exc_value is error: 

233 raise exc_type(exc_value).with_traceback(tb) 

234 else: 

235 raise error 

236 # url_for will use this result, instead of raising BuildError. 

237 return url 

238 

239 app.url_build_error_handlers.append(external_url_handler) 

240 

241 Here, `error` is the instance of :exc:`~werkzeug.routing.BuildError`, and 

242 `endpoint` and `values` are the arguments passed into `url_for`. Note 

243 that this is for building URLs outside the current application, and not for 

244 handling 404 NotFound errors. 

245 

246 .. versionadded:: 0.10 

247 The `_scheme` parameter was added. 

248 

249 .. versionadded:: 0.9 

250 The `_anchor` and `_method` parameters were added. 

251 

252 .. versionadded:: 0.9 

253 Calls :meth:`Flask.handle_build_error` on 

254 :exc:`~werkzeug.routing.BuildError`. 

255 

256 :param endpoint: the endpoint of the URL (name of the function) 

257 :param values: the variable arguments of the URL rule 

258 :param _external: if set to ``True``, an absolute URL is generated. Server 

259 address can be changed via ``SERVER_NAME`` configuration variable which 

260 falls back to the `Host` header, then to the IP and port of the request. 

261 :param _scheme: a string specifying the desired URL scheme. The `_external` 

262 parameter must be set to ``True`` or a :exc:`ValueError` is raised. The default 

263 behavior uses the same scheme as the current request, or 

264 :data:`PREFERRED_URL_SCHEME` if no request context is available. 

265 This also can be set to an empty string to build protocol-relative 

266 URLs. 

267 :param _anchor: if provided this is added as anchor to the URL. 

268 :param _method: if provided this explicitly specifies an HTTP method. 

269 """ 

270 appctx = _app_ctx_stack.top 

271 reqctx = _request_ctx_stack.top 

272 

273 if appctx is None: 

274 raise RuntimeError( 

275 "Attempted to generate a URL without the application context being" 

276 " pushed. This has to be executed when application context is" 

277 " available." 

278 ) 

279 

280 # If request specific information is available we have some extra 

281 # features that support "relative" URLs. 

282 if reqctx is not None: 

283 url_adapter = reqctx.url_adapter 

284 blueprint_name = request.blueprint 

285 

286 if endpoint[:1] == ".": 

287 if blueprint_name is not None: 

288 endpoint = f"{blueprint_name}{endpoint}" 

289 else: 

290 endpoint = endpoint[1:] 

291 

292 external = values.pop("_external", False) 

293 

294 # Otherwise go with the url adapter from the appctx and make 

295 # the URLs external by default. 

296 else: 

297 url_adapter = appctx.url_adapter 

298 

299 if url_adapter is None: 

300 raise RuntimeError( 

301 "Application was not able to create a URL adapter for request" 

302 " independent URL generation. You might be able to fix this by" 

303 " setting the SERVER_NAME config variable." 

304 ) 

305 

306 external = values.pop("_external", True) 

307 

308 anchor = values.pop("_anchor", None) 

309 method = values.pop("_method", None) 

310 scheme = values.pop("_scheme", None) 

311 appctx.app.inject_url_defaults(endpoint, values) 

312 

313 # This is not the best way to deal with this but currently the 

314 # underlying Werkzeug router does not support overriding the scheme on 

315 # a per build call basis. 

316 old_scheme = None 

317 if scheme is not None: 

318 if not external: 

319 raise ValueError("When specifying _scheme, _external must be True") 

320 old_scheme = url_adapter.url_scheme 

321 url_adapter.url_scheme = scheme 

322 

323 try: 

324 try: 

325 rv = url_adapter.build( 

326 endpoint, values, method=method, force_external=external 

327 ) 

328 finally: 

329 if old_scheme is not None: 

330 url_adapter.url_scheme = old_scheme 

331 except BuildError as error: 

332 # We need to inject the values again so that the app callback can 

333 # deal with that sort of stuff. 

334 values["_external"] = external 

335 values["_anchor"] = anchor 

336 values["_method"] = method 

337 values["_scheme"] = scheme 

338 return appctx.app.handle_url_build_error(error, endpoint, values) 

339 

340 if anchor is not None: 

341 rv += f"#{url_quote(anchor)}" 

342 return rv

367def flash(message: str, category: str = "message") -> None: 

368 """Flashes a message to the next request. In order to remove the 

369 flashed message from the session and to display it to the user, 

370 the template has to call :func:`get_flashed_messages`. 

371 

372 .. versionchanged:: 0.3 

373 `category` parameter added. 

374 

375 :param message: the message to be flashed. 

376 :param category: the category for the message. The following values 

377 are recommended: ``'message'`` for any kind of message, 

378 ``'error'`` for errors, ``'info'`` for information 

379 messages and ``'warning'`` for warnings. However any 

380 kind of string can be used as category. 

381 """ 

382 # Original implementation: 

383 # 

384 # session.setdefault('_flashes', []).append((category, message)) 

385 # 

386 # This assumed that changes made to mutable structures in the session are 

387 # always in sync with the session object, which is not true for session 

388 # implementations that use external storage for keeping their keys/values. 

389 flashes = session.get("_flashes", []) 

390 flashes.append((category, message)) 

391 session["_flashes"] = flashes 

392 message_flashed.send( 

393 current_app._get_current_object(), # type: ignore 

394 message=message, 

395 category=category, 

396 )

399def get_flashed_messages( 

400 with_categories: bool = False, category_filter: t.Iterable[str] = () 

401) -> t.Union[t.List[str], t.List[t.Tuple[str, str]]]: 

402 """Pulls all flashed messages from the session and returns them. 

403 Further calls in the same request to the function will return 

404 the same messages. By default just the messages are returned, 

405 but when `with_categories` is set to ``True``, the return value will 

406 be a list of tuples in the form ``(category, message)`` instead. 

407 

408 Filter the flashed messages to one or more categories by providing those 

409 categories in `category_filter`. This allows rendering categories in 

410 separate html blocks. The `with_categories` and `category_filter` 

411 arguments are distinct: 

412 

413 * `with_categories` controls whether categories are returned with message 

414 text (``True`` gives a tuple, where ``False`` gives just the message text). 

415 * `category_filter` filters the messages down to only those matching the 

416 provided categories. 

417 

418 See :doc:`/patterns/flashing` for examples. 

419 

420 .. versionchanged:: 0.3 

421 `with_categories` parameter added. 

422 

423 .. versionchanged:: 0.9 

424 `category_filter` parameter added. 

425 

426 :param with_categories: set to ``True`` to also receive categories. 

427 :param category_filter: filter of categories to limit return values. Only 

428 categories in the list will be returned. 

429 """ 

430 flashes = _request_ctx_stack.top.flashes 

431 if flashes is None: 

432 _request_ctx_stack.top.flashes = flashes = ( 

433 session.pop("_flashes") if "_flashes" in session else [] 

434 ) 

435 if category_filter: 

436 flashes = list(filter(lambda f: f[0] in category_filter, flashes)) 

437 if not with_categories: 

438 return [x[1] for x in flashes] 

439 return flashes

511 """Send the contents of a file to the client. 

512 

513 The first argument can be a file path or a file-like object. Paths 

514 are preferred in most cases because Werkzeug can manage the file and 

515 get extra information from the path. Passing a file-like object 

516 requires that the file is opened in binary mode, and is mostly 

517 useful when building a file in memory with :class:`io.BytesIO`. 

518 

519 Never pass file paths provided by a user. The path is assumed to be 

520 trusted, so a user could craft a path to access a file you didn't 

521 intend. Use :func:`send_from_directory` to safely serve 

522 user-requested paths from within a directory. 

523 

524 If the WSGI server sets a ``file_wrapper`` in ``environ``, it is 

525 used, otherwise Werkzeug's built-in wrapper is used. Alternatively, 

526 if the HTTP server supports ``X-Sendfile``, configuring Flask with 

527 ``USE_X_SENDFILE = True`` will tell the server to send the given 

528 path, which is much more efficient than reading it in Python. 

529 

530 :param path_or_file: The path to the file to send, relative to the 

531 current working directory if a relative path is given. 

532 Alternatively, a file-like object opened in binary mode. Make 

533 sure the file pointer is seeked to the start of the data. 

534 :param mimetype: The MIME type to send for the file. If not 

535 provided, it will try to detect it from the file name. 

536 :param as_attachment: Indicate to a browser that it should offer to 

537 save the file instead of displaying it. 

538 :param download_name: The default name browsers will use when saving 

539 the file. Defaults to the passed file name. 

540 :param conditional: Enable conditional and range responses based on 

541 request headers. Requires passing a file path and ``environ``. 

542 :param etag: Calculate an ETag for the file, which requires passing 

543 a file path. Can also be a string to use instead. 

544 :param last_modified: The last modified time to send for the file, 

545 in seconds. If not provided, it will try to detect it from the 

546 file path. 

547 :param max_age: How long the client should cache the file, in 

548 seconds. If set, ``Cache-Control`` will be ``public``, otherwise 

549 it will be ``no-cache`` to prefer conditional caching. 

550 

551 .. versionchanged:: 2.0 

552 ``download_name`` replaces the ``attachment_filename`` 

553 parameter. If ``as_attachment=False``, it is passed with 

554 ``Content-Disposition: inline`` instead. 

555 

556 .. versionchanged:: 2.0

558 ``conditional`` is enabled and ``max_age`` is not set by 

559 default. 

560 

561 .. versionchanged:: 2.0

563 string to use instead of generating one. 

564 

565 .. versionchanged:: 2.0

567 :class:`~io.TextIOBase` will raise a :exc:`ValueError` rather 

568 than sending an empty file. 

569 

570 .. versionadded:: 2.0 

571 Moved the implementation to Werkzeug. This is now a wrapper to 

572 pass some Flask-specific arguments. 

573 

574 .. versionchanged:: 1.1 

575 ``filename`` may be a :class:`~os.PathLike` object. 

576 

577 .. versionchanged:: 1.1 

578 Passing a :class:`~io.BytesIO` object supports range requests. 

579 

580 .. versionchanged:: 1.0.3

583

705def get_root_path(import_name: str) -> str: 

706 """Find the root path of a package, or the path that contains a 

707 module. If it cannot be found, returns the current working 

708 directory. 

709 

710 Not to be confused with the value returned by :func:`find_package`. 

711 

712 :meta private: 

713 """ 

714 # Module already imported and has a file attribute. Use that first. 

715 mod = sys.modules.get(import_name) 

716 

717 if mod is not None and hasattr(mod, "__file__"): 

718 return os.path.dirname(os.path.abspath(mod.__file__)) 

719 

720 # Next attempt: check the loader. 

721 loader = pkgutil.get_loader(import_name) 

722 

723 # Loader does not exist or we're referring to an unloaded main 

724 # module or a main module without path (interactive sessions), go 

725 # with the current working directory. 

726 if loader is None or import_name == "__main__": 

727 return os.getcwd() 

728 

729 if hasattr(loader, "get_filename"): 

730 filepath = loader.get_filename(import_name) # type: ignore 

731 else: 

732 # Fall back to imports. 

733 __import__(import_name) 

734 mod = sys.modules[import_name] 

735 filepath = getattr(mod, "__file__", None) 

736 

737 # If we don't have a file path it might be because it is a 

738 # namespace package. In this case pick the root path from the 

739 # first module that is contained in the package. 

740 if filepath is None: 

741 raise RuntimeError( 

742 "No root path can be found for the provided module" 

743 f" {import_name!r}. This can happen because the module" 

744 " came from an import hook that does not provide file" 

745 " name information or because it's a namespace package." 

746 " In this case the root path needs to be explicitly" 

747 " provided." 

748 ) 

749 

750 # filepath is import_name.py for a module, or __init__.py for a package. 

751 return os.path.dirname(os.path.abspath(filepath))

17# For most subplot types, a trace is associated with a particular subplot 

18# using a trace property with a name that matches the subplot type. For 

19# example, a `scatter3d.scene` property set to `'scene2'` associates a 

20# scatter3d trace with the second `scene` subplot in the figure.

22# There are a few subplot types that don't follow this pattern, and instead 

23# the trace property is just named `subplot`. For example setting 

24# the `scatterpolar.subplot` property to `polar3` associates the scatterpolar 

25# trace with the third polar subplot in the figure

171 * t (float, default 0.0): padding right of cell

533 def _check_hv_spacing(dimsize, spacing, name, dimvarname, dimname): 

534 if spacing < 0 or spacing > 1: 

535 raise ValueError("%s spacing must be between 0 and 1." % (name,)) 

536 if dimsize <= 1: 

537 return 

538 max_spacing = 1.0 / float(dimsize - 1) 

539 if spacing > max_spacing: 

540 raise ValueError( 

541 """{name} spacing cannot be greater than (1 / ({dimvarname} - 1)) = {max_spacing:f}. 

542The resulting plot would have {dimsize} {dimname} ({dimvarname}={dimsize}).""".format( 

543 dimvarname=dimvarname, 

544 name=name, 

545 dimname=dimname, 

546 max_spacing=max_spacing, 

547 dimsize=dimsize, 

548 ) 

549 )

614length {rows}.

616 rows=rows, typ=type(row_heights), val=repr(row_heights) 

617 ) 

618 ) 

619 

620 # ### column_titles / row_titles ### 

621 if column_titles and not isinstance(column_titles, (list, tuple)): 

622 raise ValueError( 

623 """ 

624The column_titles argument to make_subplots must be a list or tuple 

625 Received value of type {typ}: {val}""".format( 

626 typ=type(column_titles), val=repr(column_titles) 

627 ) 

628 ) 

629 

630 if row_titles and not isinstance(row_titles, (list, tuple)): 

631 raise ValueError( 

632 """ 

633The row_titles argument to make_subplots must be a list or tuple 

634 Received value of type {typ}: {val}""".format( 

635 typ=type(row_titles), val=repr(row_titles) 

636 ) 

637 ) 

638 

639 # Init layout 

640 # ----------- 

641 layout = go.Layout() 

642 

643 # Build grid reference 

644 # -------------------- 

645 # Built row/col sequence using 'row_dir' and 'col_dir' 

646 col_seq = range(cols)[::col_dir] 

647 row_seq = range(rows)[::row_dir] 

648 

649 # Build 2D array of tuples of the start x and start y coordinate of each 

650 # subplot 

651 grid = [ 

652 [ 

653 ( 

654 (sum(widths[:c]) + c * horizontal_spacing), 

655 (sum(heights[:r]) + r * vertical_spacing), 

656 ) 

657 for c in col_seq 

658 ] 

659 for r in row_seq 

660 ] 

661 

662 domains_grid = [[None for _ in range(cols)] for _ in range(rows)] 

663 

664 # Initialize subplot reference lists for the grid and insets 

665 grid_ref = [[None for c in range(cols)] for r in range(rows)] 

666 

667 list_of_domains = [] # added for subplot titles 

668 

669 max_subplot_ids = _get_initial_max_subplot_ids() 

670 

671 # Loop through specs -- (r, c) <-> (row, col) 

672 for r, spec_row in enumerate(specs): 

673 for c, spec in enumerate(spec_row): 

674 

675 if spec is None: # skip over None cells 

676 continue 

677 

678 # ### Compute x and y domain for subplot ### 

679 c_spanned = c + spec["colspan"] - 1 # get spanned c 

680 r_spanned = r + spec["rowspan"] - 1 # get spanned r 

681 

682 # Throw exception if 'colspan' | 'rowspan' is too large for grid 

683 if c_spanned >= cols: 

684 raise Exception( 

685 "Some 'colspan' value is too large for " "this subplot grid." 

686 ) 

687 if r_spanned >= rows: 

688 raise Exception( 

689 "Some 'rowspan' value is too large for " "this subplot grid." 

690 ) 

691 

692 # Get x domain using grid and colspan 

693 x_s = grid[r][c][0] + spec["l"] 

694 

695 x_e = grid[r][c_spanned][0] + widths[c_spanned] - spec["r"] 

696 x_domain = [x_s, x_e] 

697 

698 # Get y domain (dep. on row_dir) using grid & r_spanned 

699 if row_dir > 0: 

700 y_s = grid[r][c][1] + spec["b"] 

701 y_e = grid[r_spanned][c][1] + heights[r_spanned] - spec["t"] 

702 else: 

703 y_s = grid[r_spanned][c][1] + spec["b"] 

704 y_e = grid[r][c][1] + heights[-1 - r] - spec["t"] 

705 y_domain = [y_s, y_e] 

706 

707 list_of_domains.append(x_domain) 

708 list_of_domains.append(y_domain) 

709 

710 domains_grid[r][c] = [x_domain, y_domain] 

711 

712 # ### construct subplot container ### 

713 subplot_type = spec["type"] 

714 secondary_y = spec["secondary_y"] 

715 subplot_refs = _init_subplot( 

716 layout, subplot_type, secondary_y, x_domain, y_domain, max_subplot_ids 

717 ) 

718 grid_ref[r][c] = subplot_refs 

719 

720 _configure_shared_axes(layout, grid_ref, specs, "x", shared_xaxes, row_dir) 

721 _configure_shared_axes(layout, grid_ref, specs, "y", shared_yaxes, row_dir) 

722 

723 # Build inset reference 

724 # --------------------- 

725 # Loop through insets 

726 insets_ref = [None for inset in range(len(insets))] if insets else None 

727 if insets: 

728 for i_inset, inset in enumerate(insets): 

729 

730 r = inset["cell"][0] - 1 

731 c = inset["cell"][1] - 1 

732 

733 # Throw exception if r | c is out of range 

734 if not (0 <= r < rows): 

735 raise Exception( 

736 "Some 'cell' row value is out of range. " 

737 "Note: the starting cell is (1, 1)" 

738 ) 

739 if not (0 <= c < cols): 

740 raise Exception( 

741 "Some 'cell' col value is out of range. " 

742 "Note: the starting cell is (1, 1)" 

743 ) 

744 

745 # Get inset x domain using grid 

746 x_s = grid[r][c][0] + inset["l"] * widths[c] 

747 if inset["w"] == "to_end": 

748 x_e = grid[r][c][0] + widths[c] 

749 else: 

750 x_e = x_s + inset["w"] * widths[c] 

751 x_domain = [x_s, x_e] 

752 

753 # Get inset y domain using grid 

754 y_s = grid[r][c][1] + inset["b"] * heights[-1 - r] 

755 if inset["h"] == "to_end": 

756 y_e = grid[r][c][1] + heights[-1 - r] 

757 else: 

758 y_e = y_s + inset["h"] * heights[-1 - r] 

759 y_domain = [y_s, y_e] 

760 

761 list_of_domains.append(x_domain) 

762 list_of_domains.append(y_domain) 

763 

764 subplot_type = inset["type"] 

765 

766 subplot_refs = _init_subplot( 

767 layout, subplot_type, False, x_domain, y_domain, max_subplot_ids 

768 ) 

769 

770 insets_ref[i_inset] = subplot_refs 

771 

772 # Build grid_str 

773 # This is the message printed when print_grid=True 

774 grid_str = _build_grid_str(specs, grid_ref, insets, insets_ref, row_seq) 

775 

776 # Add subplot titles 

777 plot_title_annotations = _build_subplot_title_annotations( 

778 subplot_titles, list_of_domains 

779 ) 

780 

781 layout["annotations"] = plot_title_annotations 

782 

783 # Add column titles 

784 if column_titles: 

785 domains_list = [] 

786 if row_dir > 0: 

787 for c in range(cols): 

788 domain_pair = domains_grid[-1][c] 

789 if domain_pair: 

790 domains_list.extend(domain_pair) 

791 else: 

792 for c in range(cols): 

793 domain_pair = domains_grid[0][c] 

794 if domain_pair: 

795 domains_list.extend(domain_pair) 

796 

797 # Add subplot titles 

798 column_title_annotations = _build_subplot_title_annotations( 

799 column_titles, domains_list 

800 ) 

801 

802 layout["annotations"] += tuple(column_title_annotations) 

803 

804 if row_titles: 

805 domains_list = [] 

806 for r in range(rows): 

807 domain_pair = domains_grid[r][-1] 

808 if domain_pair: 

809 domains_list.extend(domain_pair) 

810 

811 # Add subplot titles 

812 column_title_annotations = _build_subplot_title_annotations( 

813 row_titles, domains_list, title_edge="right" 

814 ) 

815 

816 layout["annotations"] += tuple(column_title_annotations) 

817 

818 if x_title: 

819 domains_list = [(0, max_width), (0, 1)] 

820 

821 # Add subplot titles 

822 column_title_annotations = _build_subplot_title_annotations( 

823 [x_title], domains_list, title_edge="bottom", offset=30 

824 ) 

825 

826 layout["annotations"] += tuple(column_title_annotations) 

827 

828 if y_title: 

829 domains_list = [(0, 1), (0, 1)] 

830 

831 # Add subplot titles 

832 column_title_annotations = _build_subplot_title_annotations( 

833 [y_title], domains_list, title_edge="left", offset=40 

834 ) 

835 

836 layout["annotations"] += tuple(column_title_annotations) 

837 

838 # Handle displaying grid information 

839 if print_grid: 

840 print(grid_str) 

841 

842 # Build resulting figure 

843 if figure is None: 

844 figure = go.Figure() 

845 figure.update_layout(layout) 

846 

847 # Attach subplot grid info to the figure 

848 figure.__dict__["_grid_ref"] = grid_ref 

849 figure.__dict__["_grid_str"] = grid_str 

850 

851 return figure

855 rows = len(grid_ref)

865 def update_axis_matches(first_axis_id, subplot_ref, spec, remove_label): 

866 if subplot_ref is None: 

867 return first_axis_id 

868 

869 if x_or_y == "x": 

870 span = spec["colspan"] 

871 else: 

872 span = spec["rowspan"] 

873 

874 if subplot_ref.subplot_type == "xy" and span == 1: 

875 if first_axis_id is None: 

876 first_axis_name = subplot_ref.layout_keys[layout_key_ind]

878 else:

881 axis_to_match.matches = first_axis_id 

882 if remove_label: 

883 axis_to_match.showticklabels = False 

884 

885 return first_axis_id 

886 

887 if shared == "columns" or (x_or_y == "x" and shared is True): 

888 for c in range(cols):

890 ok_to_remove_label = x_or_y == "x"

899 

900 elif shared == "rows" or (x_or_y == "y" and shared is True): 

901 for r in rows_iter: 

902 first_axis_id = None 

903 ok_to_remove_label = x_or_y == "y" 

904 for c in range(cols): 

905 if not grid_ref[r][c]: 

906 continue 

907 subplot_ref = grid_ref[r][c][0] 

908 spec = specs[r][c] 

909 first_axis_id = update_axis_matches( 

910 first_axis_id, subplot_ref, spec, ok_to_remove_label 

911 ) 

912 

913 elif shared == "all": 

914 first_axis_id = None 

915 for c in range(cols): 

916 for ri, r in enumerate(rows_iter): 

917 if not grid_ref[r][c]: 

918 continue 

919 subplot_ref = grid_ref[r][c][0] 

920 spec = specs[r][c] 

921 

922 if x_or_y == "y": 

923 ok_to_remove_label = c > 0 

924 else: 

925 ok_to_remove_label = ri > 0 if row_dir > 0 else r < rows - 1 

926 

927 first_axis_id = update_axis_matches( 

928 first_axis_id, subplot_ref, spec, ok_to_remove_label 

929 )

1028def _subplot_type_for_trace_type(trace_type): 

1029 from plotly.validators import DataValidator 

1030 

1031 trace_validator = DataValidator() 

1032 if trace_type in trace_validator.class_strs_map: 

1033 # subplot_type is a trace name, find the subplot type for trace 

1034 trace = trace_validator.validate_coerce([{"type": trace_type}])[0] 

1035 if "domain" in trace: 

1036 return "domain" 

1037 elif "xaxis" in trace and "yaxis" in trace: 

1038 return "xy" 

1039 elif "geo" in trace: 

1040 return "geo" 

1041 elif "scene" in trace: 

1042 return "scene" 

1043 elif "subplot" in trace: 

1044 for t in _subplot_prop_named_subplot: 

1045 try: 

1046 trace.subplot = t 

1047 return t 

1048 except ValueError: 

1049 pass 

1050 

1051 return None

1054def _validate_coerce_subplot_type(subplot_type): 

1055 

1056 # Lowercase subplot_type 

1057 orig_subplot_type = subplot_type 

1058 subplot_type = subplot_type.lower() 

1059 

1060 # Check if it's a named subplot type 

1061 if subplot_type in _subplot_types: 

1062 return subplot_type 

1063 

1064 # Try to determine subplot type for trace 

1065 subplot_type = _subplot_type_for_trace_type(subplot_type) 

1066 

1067 if subplot_type is None: 

1068 raise ValueError("Unsupported subplot type: {}".format(repr(orig_subplot_type))) 

1069 else: 

1070 return subplot_type

1110def _build_subplot_title_annotations( 

1111 subplot_titles, list_of_domains, title_edge="top", offset=0 

1112): 

1113 # If shared_axes is False (default) use list_of_domains 

1114 # This is used for insets and irregular layouts 

1115 # if not shared_xaxes and not shared_yaxes: 

1116 x_dom = list_of_domains[::2] 

1117 y_dom = list_of_domains[1::2] 

1118 subtitle_pos_x = [] 

1119 subtitle_pos_y = [] 

1120 

1121 if title_edge == "top": 

1122 text_angle = 0 

1123 xanchor = "center" 

1124 yanchor = "bottom" 

1125 

1126 for x_domains in x_dom: 

1127 subtitle_pos_x.append(sum(x_domains) / 2.0) 

1128 for y_domains in y_dom: 

1129 subtitle_pos_y.append(y_domains[1]) 

1130 

1131 yshift = offset 

1132 xshift = 0 

1133 elif title_edge == "bottom": 

1134 text_angle = 0 

1135 xanchor = "center" 

1136 yanchor = "top" 

1137 

1138 for x_domains in x_dom: 

1139 subtitle_pos_x.append(sum(x_domains) / 2.0) 

1140 for y_domains in y_dom: 

1141 subtitle_pos_y.append(y_domains[0]) 

1142 

1143 yshift = -offset 

1144 xshift = 0 

1145 elif title_edge == "right": 

1146 text_angle = 90 

1147 xanchor = "left" 

1148 yanchor = "middle" 

1149 

1150 for x_domains in x_dom: 

1151 subtitle_pos_x.append(x_domains[1]) 

1152 for y_domains in y_dom: 

1153 subtitle_pos_y.append(sum(y_domains) / 2.0) 

1154 

1155 yshift = 0 

1156 xshift = offset 

1157 elif title_edge == "left": 

1158 text_angle = -90 

1159 xanchor = "right" 

1160 yanchor = "middle" 

1161 

1162 for x_domains in x_dom: 

1163 subtitle_pos_x.append(x_domains[0]) 

1164 for y_domains in y_dom: 

1165 subtitle_pos_y.append(sum(y_domains) / 2.0) 

1166 

1167 yshift = 0 

1168 xshift = -offset 

1169 else: 

1170 raise ValueError("Invalid annotation edge '{edge}'".format(edge=title_edge)) 

1171 

1172 plot_titles = [] 

1173 for index in range(len(subplot_titles)): 

1174 if not subplot_titles[index] or index >= len(subtitle_pos_y): 

1175 pass 

1176 else: 

1177 annot = { 

1178 "y": subtitle_pos_y[index], 

1179 "xref": "paper", 

1180 "x": subtitle_pos_x[index], 

1181 "yref": "paper", 

1182 "text": subplot_titles[index], 

1183 "showarrow": False, 

1184 "font": dict(size=16), 

1185 "xanchor": xanchor, 

1186 "yanchor": yanchor, 

1187 } 

1188 

1189 if xshift != 0: 

1190 annot["xshift"] = xshift 

1191 

1192 if yshift != 0: 

1193 annot["yshift"] = yshift 

1194 

1195 if text_angle != 0: 

1196 annot["textangle"] = text_angle 

1197 

1198 plot_titles.append(annot) 

1199 return plot_titles

1203

1216

1232 layout_keys = sorted({k for ref in subplot_refs for k in ref.layout_keys})

1288 else: 

1289 cell_str += padding + e_str 

1290 

1291 if spec["rowspan"] > 1: 

1292 for cc in range(spec["colspan"]): 

1293 for rr in range(1, spec["rowspan"]):

1300

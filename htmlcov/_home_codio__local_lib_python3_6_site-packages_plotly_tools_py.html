75def mpl_to_plotly(fig, resize=False, strip_style=False, verbose=False): 

76 """Convert a matplotlib figure to plotly dictionary and send. 

77 

78 All available information about matplotlib visualizations are stored 

79 within a matplotlib.figure.Figure object. You can create a plot in python 

80 using matplotlib, store the figure object, and then pass this object to 

81 the fig_to_plotly function. In the background, mplexporter is used to 

82 crawl through the mpl figure object for appropriate information. This 

83 information is then systematically sent to the PlotlyRenderer which 

84 creates the JSON structure used to make plotly visualizations. Finally, 

85 these dictionaries are sent to plotly and your browser should open up a 

86 new tab for viewing! Optionally, if you're working in IPython, you can 

87 set notebook=True and the PlotlyRenderer will call plotly.iplot instead 

88 of plotly.plot to have the graph appear directly in the IPython notebook. 

89 

90 Note, this function gives the user access to a simple, one-line way to 

91 render an mpl figure in plotly. If you need to trouble shoot, you can do 

92 this step manually by NOT running this fuction and entereing the following: 

93 

94 =========================================================================== 

95 from plotly.matplotlylib import mplexporter, PlotlyRenderer 

96 

97 # create an mpl figure and store it under a varialble 'fig' 

98 

99 renderer = PlotlyRenderer() 

100 exporter = mplexporter.Exporter(renderer) 

101 exporter.run(fig) 

102 =========================================================================== 

103 

104 You can then inspect the JSON structures by accessing these: 

105 

106 renderer.layout -- a plotly layout dictionary 

107 renderer.data -- a list of plotly data dictionaries 

108 """ 

109 matplotlylib = optional_imports.get_module("plotly.matplotlylib") 

110 if matplotlylib: 

111 renderer = matplotlylib.PlotlyRenderer() 

112 matplotlylib.Exporter(renderer).run(fig) 

113 if resize: 

114 renderer.resize() 

115 if strip_style: 

116 renderer.strip_style() 

117 if verbose: 

118 print(renderer.msg) 

119 return renderer.plotly_fig 

120 else: 

121 warnings.warn( 

122 "To use Plotly's matplotlylib functionality, you'll need to have " 

123 "matplotlib successfully installed with all of its dependencies. " 

124 "You're getting this error because matplotlib or one of its " 

125 "dependencies doesn't seem to be installed correctly."

132def get_subplots(rows=1, columns=1, print_grid=False, **kwargs): 

133 """Return a dictionary instance with the subplots set in 'layout'. 

134 

135 Example 1: 

136 # stack two subplots vertically 

137 fig = tools.get_subplots(rows=2) 

138 fig['data'] += [Scatter(x=[1,2,3], y=[2,1,2], xaxis='x1', yaxis='y1')] 

139 fig['data'] += [Scatter(x=[1,2,3], y=[2,1,2], xaxis='x2', yaxis='y2')] 

140 

141 Example 2: 

142 # print out string showing the subplot grid you've put in the layout 

143 fig = tools.get_subplots(rows=3, columns=2, print_grid=True) 

144 

145 Keywords arguments with constant defaults: 

146 

147 rows (kwarg, int greater than 0, default=1): 

148 Number of rows, evenly spaced vertically on the figure. 

149 

150 columns (kwarg, int greater than 0, default=1): 

151 Number of columns, evenly spaced horizontally on the figure. 

152 

153 horizontal_spacing (kwarg, float in [0,1], default=0.1): 

154 Space between subplot columns. Applied to all columns. 

155 

156 vertical_spacing (kwarg, float in [0,1], default=0.05): 

157 Space between subplot rows. Applied to all rows. 

158 

159 print_grid (kwarg, True | False, default=False): 

160 If True, prints a tab-delimited string representation 

161 of your plot grid. 

162 

163 Keyword arguments with variable defaults: 

164 

165 horizontal_spacing (kwarg, float in [0,1], default=0.2 / columns): 

166 Space between subplot columns. 

167 

168 vertical_spacing (kwarg, float in [0,1], default=0.3 / rows): 

169 Space between subplot rows. 

170 

171 """ 

172 # TODO: protected until #282 

173 from plotly.graph_objs import graph_objs 

174 

175 warnings.warn( 

176 "tools.get_subplots is depreciated. " "Please use tools.make_subplots instead." 

177 ) 

178 

179 # Throw exception for non-integer rows and columns 

180 if not isinstance(rows, int) or rows <= 0: 

181 raise Exception("Keyword argument 'rows' " "must be an int greater than 0") 

182 if not isinstance(columns, int) or columns <= 0: 

183 raise Exception("Keyword argument 'columns' " "must be an int greater than 0") 

184 

185 # Throw exception if non-valid kwarg is sent 

186 VALID_KWARGS = ["horizontal_spacing", "vertical_spacing"] 

187 for key in kwargs.keys(): 

188 if key not in VALID_KWARGS: 

189 raise Exception("Invalid keyword argument: '{0}'".format(key)) 

190 

191 # Set 'horizontal_spacing' / 'vertical_spacing' w.r.t. rows / columns 

192 try: 

193 horizontal_spacing = float(kwargs["horizontal_spacing"]) 

194 except KeyError: 

195 horizontal_spacing = 0.2 / columns 

196 try: 

197 vertical_spacing = float(kwargs["vertical_spacing"]) 

198 except KeyError: 

199 vertical_spacing = 0.3 / rows 

200 

201 fig = dict(layout=graph_objs.Layout()) # will return this at the end 

202 plot_width = (1 - horizontal_spacing * (columns - 1)) / columns 

203 plot_height = (1 - vertical_spacing * (rows - 1)) / rows 

204 plot_num = 0 

205 for rrr in range(rows): 

206 for ccc in range(columns): 

207 xaxis_name = "xaxis{0}".format(plot_num + 1) 

208 x_anchor = "y{0}".format(plot_num + 1) 

209 x_start = (plot_width + horizontal_spacing) * ccc 

210 x_end = x_start + plot_width 

211 

212 yaxis_name = "yaxis{0}".format(plot_num + 1) 

213 y_anchor = "x{0}".format(plot_num + 1) 

214 y_start = (plot_height + vertical_spacing) * rrr 

215 y_end = y_start + plot_height 

216 

217 xaxis = dict(domain=[x_start, x_end], anchor=x_anchor) 

218 fig["layout"][xaxis_name] = xaxis 

219 yaxis = dict(domain=[y_start, y_end], anchor=y_anchor) 

220 fig["layout"][yaxis_name] = yaxis 

221 plot_num += 1 

222 

223 if print_grid: 

224 print("This is the format of your plot grid!") 

225 grid_string = "" 

226 plot = 1 

227 for rrr in range(rows): 

228 grid_line = "" 

229 for ccc in range(columns): 

230 grid_line += "[{0}]\t".format(plot) 

231 plot += 1 

232 grid_string = grid_line + "\n" + grid_string 

233 print(grid_string) 

234 

235 return graph_objs.Figure(fig) # forces us to validate what we just did...

503def _replace_newline(obj): 

504 """Replaces '\n' with '<br>' for all strings in a collection.""" 

505 if isinstance(obj, dict): 

506 d = dict() 

507 for key, val in list(obj.items()): 

508 d[key] = _replace_newline(val) 

509 return d 

510 elif isinstance(obj, list): 

511 l = list() 

512 for index, entry in enumerate(obj): 

513 l += [_replace_newline(entry)] 

514 return l 

515 elif isinstance(obj, six.string_types): 

516 s = obj.replace("\n", "<br>") 

517 if s != obj: 

518 warnings.warn( 

519 "Looks like you used a newline character: '\\n'.\n\n" 

520 "Plotly uses a subset of HTML escape characters\n" 

521 "to do things like newline (<br>), bold (<b></b>),\n" 

522 "italics (<i></i>), etc. Your newline characters \n" 

523 "have been converted to '<br>' so they will show \n" 

524 "up right on your Plotly figure!" 

525 ) 

526 return s 

527 else: 

528 return obj # we return the actual reference... but DON'T mutate.

531def return_figure_from_figure_or_data(figure_or_data, validate_figure): 

532 from plotly.graph_objs import Figure 

533 from plotly.basedatatypes import BaseFigure 

534 

535 validated = False 

536 if isinstance(figure_or_data, dict): 

537 figure = figure_or_data 

538 elif isinstance(figure_or_data, list): 

539 figure = {"data": figure_or_data} 

540 elif isinstance(figure_or_data, BaseFigure): 

541 figure = figure_or_data.to_dict() 

542 validated = True 

543 else: 

544 raise exceptions.PlotlyError( 

545 "The `figure_or_data` positional " 

546 "argument must be " 

547 "`dict`-like, `list`-like, or an instance of plotly.graph_objs.Figure" 

548 ) 

549 

550 if validate_figure and not validated: 

551 

552 try: 

553 figure = Figure(**figure).to_dict() 

554 except exceptions.PlotlyError as err: 

555 raise exceptions.PlotlyError( 

556 "Invalid 'figure_or_data' argument. " 

557 "Plotly will not be able to properly " 

558 "parse the resulting JSON. If you " 

559 "want to send this 'figure_or_data' " 

560 "to Plotly anyway (not recommended), " 

561 "you can set 'validate=False' as a " 

562 "plot option.\nHere's why you're " 

563 "seeing this error:\n\n{0}" 

564 "".format(err) 

565 ) 

566 if not figure["data"]: 

567 raise exceptions.PlotlyEmptyDataError( 

568 "Empty data list found. Make sure that you populated the " 

569 "list of data objects you're sending and try again.\n" 

570 "Questions? Visit support.plot.ly" 

571 ) 

572 

573 return figure

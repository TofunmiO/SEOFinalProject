42class options: 

43 """The `options` object contains default configuration values for 

44 geocoders, e.g. `timeout` and `User-Agent`. 

45 Instead of passing a custom value to each geocoder individually, you can 

46 override a default value in this object. 

47 

48 Please note that not all geocoders use all attributes of this object. 

49 For example, some geocoders don't respect the ``default_scheme`` 

50 attribute. Refer to the specific geocoder's initializer doc for a list 

51 of parameters which that geocoder accepts. 

52 

53 Example for overriding default ``timeout`` and ``user_agent``:: 

54 

55 >>> import geopy.geocoders 

56 >>> from geopy.geocoders import Nominatim 

57 >>> geopy.geocoders.options.default_user_agent = 'my_app/1' 

58 >>> geopy.geocoders.options.default_timeout = 7 

59 >>> geolocator = Nominatim() 

60 >>> print(geolocator.headers) 

61 {'User-Agent': 'my_app/1'} 

62 >>> print(geolocator.timeout) 

63 7

66 default_adapter_factory 

67 A callable which returns a :class:`geopy.adapters.BaseAdapter` 

68 instance. Adapters are different implementations of HTTP clients. 

69 See :mod:`geopy.adapters` for more info. 

70 

71 This callable accepts two keyword args: ``proxies`` and ``ssl_context``. 

72 A class might be specified as this callable as well. 

73 

74 Example:: 

75 

76 import geopy.geocoders 

77 geopy.geocoders.options.default_adapter_factory \ 

78= geopy.adapters.URLLibAdapter 

79 

80 geopy.geocoders.options.default_adapter_factory = ( 

81 lambda proxies, ssl_context: MyAdapter( 

82 proxies=proxies, ssl_context=ssl_context, my_custom_arg=42 

83 ) 

84 ) 

85 

86 If `requests <https://requests.readthedocs.io>`_ package is 

87 installed, the default adapter is 

88 :class:`geopy.adapters.RequestsAdapter`. Otherwise it is 

89 :class:`geopy.adapters.URLLibAdapter`. 

90 

91 .. versionadded:: 2.0

93 default_proxies 

94 Tunnel requests through HTTP proxy. 

95 

96 By default the system proxies are respected (e.g. 

97 `HTTP_PROXY` and `HTTPS_PROXY` env vars or platform-specific 

98 proxy settings, such as macOS or Windows native 

99 preferences -- see :func:`urllib.request.getproxies` for 

100 more details). The `proxies` value for using system proxies 

101 is ``None``. 

102 

103 To disable system proxies and issue requests directly, 

104 explicitly pass an empty dict as a value for `proxies`: ``{}``. 

105 

106 To use a custom HTTP proxy location, pass a string. 

107 Valid examples are: 

108 

109 - ``"192.0.2.0:8080"`` 

110 - ``"john:passw0rd@192.0.2.0:8080"`` 

111 - ``"http://john:passw0rd@192.0.2.0:8080"`` 

112 

113 Please note: 

114 

115 - Scheme part (``http://``) of the proxy is ignored. 

116 - Only `http` proxy is supported. Even if the proxy scheme 

117 is `https`, it will be ignored, and the connection between 

118 client and proxy would still be unencrypted. 

119 However, `https` requests via `http` proxy are still 

120 supported (via `HTTP CONNECT` method). 

121 

122 

123 Raw urllib-style `proxies` dict might be provided instead of 

124 a string: 

125 

126 - ``{"https": "192.0.2.0:8080"}`` -- means that HTTP proxy 

127 would be used only for requests having `https` scheme. 

128 String `proxies` value is automatically used for both 

129 schemes, and is provided as a shorthand for the urllib-style 

130 `proxies` dict. 

131 

132 For more information, see 

133 documentation on :func:`urllib.request.getproxies`. 

134 

135 default_scheme 

136 Use ``'https'`` or ``'http'`` as the API URL's scheme. 

137 

138 default_ssl_context 

139 An :class:`ssl.SSLContext` instance with custom TLS 

140 verification settings. Pass ``None`` to use the interpreter's 

141 defaults (that is to use the system's trusted CA certificates). 

142 

143 To use the CA bundle used by `requests` library:: 

144 

145 import ssl 

146 import certifi 

147 import geopy.geocoders 

148 ctx = ssl.create_default_context(cafile=certifi.where()) 

149 geopy.geocoders.options.default_ssl_context = ctx 

150 

151 To disable TLS certificate verification completely:: 

152 

153 import ssl 

154 import geopy.geocoders 

155 ctx = ssl.create_default_context() 

156 ctx.check_hostname = False 

157 ctx.verify_mode = ssl.CERT_NONE 

158 geopy.geocoders.options.default_ssl_context = ctx 

159 

160 See docs for the :class:`ssl.SSLContext` class for more examples. 

161

294 def _coerce_point_to_string(self, point, output_format="%(lat)s,%(lon)s"): 

295 """ 

296 Do the right thing on "point" input. For geocoders with reverse 

297 methods. 

298 """ 

299 if not isinstance(point, Point): 

300 point = Point(point) 

301 

302 # Altitude is silently dropped. 

303 # 

304 # Geocoding services (almost?) always consider only lat and lon 

305 # in queries, so altitude doesn't affect the request. 

306 # A non-zero altitude should not raise an exception 

307 # though, because PoIs are assumed to span the whole 

308 # altitude axis (i.e. not just the 0km plane). 

309 return output_format % dict(lat=point.latitude, 

310 lon=point.longitude)

312 def _format_bounding_box( 

313 self, bbox, output_format="%(lat1)s,%(lon1)s,%(lat2)s,%(lon2)s" 

314 ): 

315 """ 

316 Transform bounding box boundaries to a string matching 

317 `output_format` from the following formats: 

318 

319 - [Point(lat1, lon1), Point(lat2, lon2)] 

320 - [[lat1, lon1], [lat2, lon2]] 

321 - ["lat1,lon1", "lat2,lon2"] 

322 

323 It is guaranteed that lat1 <= lat2 and lon1 <= lon2. 

324 """ 

325 if len(bbox) != 2: 

326 raise GeocoderQueryError("Unsupported format for a bounding box") 

327 p1, p2 = bbox 

328 p1, p2 = Point(p1), Point(p2) 

329 return output_format % dict(lat1=min(p1.latitude, p2.latitude), 

330 lon1=min(p1.longitude, p2.longitude), 

331 lat2=max(p1.latitude, p2.latitude), 

332 lon2=max(p1.longitude, p2.longitude))

424def _synchronized(func): 

425 """A decorator for geocoder methods which makes the method always run 

426 under a lock. The lock is reentrant. 

427 

428 This decorator transparently handles sync and async working modes. 

429 """ 

430 

431 sync_lock = threading.RLock() 

432 

433 def locked_sync(self, *args, **kwargs): 

434 with sync_lock: 

435 return func(self, *args, **kwargs) 

436 

437 # At the moment this decorator is evaluated we don't know if we 

438 # will work in sync or async mode. 

439 # But we shouldn't create the asyncio Lock in sync mode to avoid 

440 # unwanted implicit loop initialization. 

441 async_lock = None # asyncio.Lock() 

442 async_lock_task = None # support reentrance 

443 

444 async def locked_async(self, *args, **kwargs): 

445 nonlocal async_lock 

446 nonlocal async_lock_task 

447 

448 if async_lock is None: 

449 async_lock = asyncio.Lock() 

450 

451 if async_lock.locked(): 

452 assert async_lock_task is not None 

453 if compat.current_task() is async_lock_task: 

454 res = func(self, *args, **kwargs) 

455 if inspect.isawaitable(res): 

456 res = await res 

457 return res 

458 

459 async with async_lock: 

460 async_lock_task = compat.current_task() 

461 try: 

462 res = func(self, *args, **kwargs) 

463 if inspect.isawaitable(res): 

464 res = await res 

465 return res 

466 finally: 

467 async_lock_task = None 

468 

469 @functools.wraps(func) 

470 def f(self, *args, **kwargs): 

471 run_async = isinstance(self.adapter, BaseAsyncAdapter) 

472 if run_async: 

473 return locked_async(self, *args, **kwargs) 

474 else: 

475 return locked_sync(self, *args, **kwargs) 

476 

477 return f

127 def _get_signed_url(self, params): 

128 """ 

129 Returns a Premier account signed url. Docs on signature: 

130 https://developers.google.com/maps/documentation/business/webservices/auth#digital_signatures 

131 """ 

132 params['client'] = self.client_id 

133 

134 if self.channel: 

135 params['channel'] = self.channel 

136 

137 path = "?".join((self.api_path, urlencode(params))) 

138 signature = hmac.new( 

139 base64.urlsafe_b64decode(self.secret_key), 

140 path.encode('utf-8'), 

141 hashlib.sha1 

142 ) 

143 signature = base64.urlsafe_b64encode( 

144 signature.digest() 

145 ).decode('utf-8') 

146 return '%s://%s%s&signature=%s' % ( 

147 self.scheme, self.domain, path, signature

151 """ 

152 Format the components dict to something Google understands. 

153 """ 

154 component_items = [] 

155 

156 if isinstance(components, collections.abc.Mapping): 

157 component_items = components.items()

182 sensor=False

184 """ 

185 Return a location point by address. 

186 

187 :param str query: The address or query you wish to geocode. Optional, 

188 if ``components`` param is set:: 

189 

190 >>> g.geocode(components={"city": "Paris", "country": "FR"}) 

191 Location(France, (46.227638, 2.213749, 0.0)) 

192 

193 :param bool exactly_one: Return one result or a list of results, if 

194 available. 

195 

196 :param int timeout: Time, in seconds, to wait for the geocoding service 

197 to respond before raising a :class:`geopy.exc.GeocoderTimedOut` 

198 exception. Set this only if you wish to override, on this call 

199 only, the value set during the geocoder's initialization. 

200 

201 :type bounds: list or tuple of 2 items of :class:`geopy.point.Point` or 

202 ``(latitude, longitude)`` or ``"%(latitude)s, %(longitude)s"``. 

203 :param bounds: The bounding box of the viewport within which 

204 to bias geocode results more prominently. 

205 Example: ``[Point(22, 180), Point(-22, -180)]``. 

206 

207 :param str region: The region code, specified as a ccTLD 

208 ("top-level domain") two-character value. 

209 

210 :type components: dict or list 

211 :param components: Restricts to an area. Can use any combination of: 

212 `route`, `locality`, `administrative_area`, `postal_code`, 

213 `country`. 

214 

215 Pass a list of tuples if you want to specify multiple components of 

216 the same type, e.g.: 

217 

218 >>> [('administrative_area', 'VA'), ('administrative_area', 'Arlington')] 

219 

220 :param str place_id: Retrieve a Location using a Place ID. 

221 Cannot be not used with ``query`` or ``bounds`` parameters. 

222 

223 >>> g.geocode(place_id='ChIJOcfP0Iq2j4ARDrXUa7ZWs34') 

224 

225 :param str language: The language in which to return results. 

226 

227 :param bool sensor: Whether the geocoding request comes from a 

228 device with a location sensor. 

229 

230 :rtype: ``None``, :class:`geopy.location.Location` or a list of them, if 

231 ``exactly_one=False``. 

232 """ 

233 params = { 

234 'sensor': str(sensor).lower() 

235 } 

236 if place_id and (bounds or query): 

237 raise ValueError( 

238 'Only one of the `query` or `place id` or `bounds` ' 

239 ' parameters must be entered.') 

240 

241 if place_id is not None: 

242 params['place_id'] = place_id 

243 

244 if query is not None: 

245 params['address'] = query 

246 

247 if query is None and place_id is None and not components: 

248 raise ValueError('Either `query` or `components` or `place_id` ' 

249 'must be set.') 

250 

251 if self.api_key: 

252 params['key'] = self.api_key 

253 if bounds: 

254 params['bounds'] = self._format_bounding_box( 

255 bounds, "%(lat1)s,%(lon1)s|%(lat2)s,%(lon2)s") 

256 if region: 

257 params['region'] = region 

258 if components: 

259 params['components'] = self._format_components_param(components) 

260 if language: 

261 params['language'] = language 

262 

263 if self.premier: 

264 url = self._get_signed_url(params) 

265 else: 

266 url = "?".join((self.api, urlencode(params))) 

267 

268 logger.debug("%s.geocode: %s", self.__class__.__name__, url) 

269 callback = partial(self._parse_json, exactly_one=exactly_one) 

270 return self._call_geocoder(url, callback, timeout=timeout)

324 def reverse_timezone(self, query, *, at_time=None, timeout=DEFAULT_SENTINEL): 

325 """ 

326 Find the timezone a point in `query` was in for a specified `at_time`. 

327 

328 `None` will be returned for points without an assigned 

329 Olson timezone id (e.g. for Antarctica). 

330 

331 :param query: The coordinates for which you want a timezone. 

332 :type query: :class:`geopy.point.Point`, list or tuple of (latitude, 

333 longitude), or string as "%(latitude)s, %(longitude)s" 

334 

335 :param at_time: The time at which you want the timezone of this 

336 location. This is optional, and defaults to the time that the 

337 function is called in UTC. Timezone-aware datetimes are correctly 

338 handled and naive datetimes are silently treated as UTC. 

339 :type at_time: :class:`datetime.datetime` or None 

340 

341 :param int timeout: Time, in seconds, to wait for the geocoding service 

342 to respond before raising a :class:`geopy.exc.GeocoderTimedOut` 

343 exception. Set this only if you wish to override, on this call 

344 only, the value set during the geocoder's initialization. 

345 

346 :rtype: ``None`` or :class:`geopy.timezone.Timezone`. 

347 """ 

348 ensure_pytz_is_installed()

351 

352 timestamp = self._normalize_timezone_at_time(at_time) 

353 

354 params = { 

355 "location": location, 

356 "timestamp": timestamp, 

357 } 

358 if self.api_key: 

359 params['key'] = self.api_key 

360 url = "?".join((self.tz_api, urlencode(params)))

365 def _parse_json_timezone(self, response): 

366 status = response.get('status') 

367 if status != 'OK': 

368 self._check_status(status) 

369 

370 timezone_id = response.get("timeZoneId") 

371 if timezone_id is None: 

372 # Google returns `status: ZERO_RESULTS` for uncovered 

373 # points (e.g. for Antarctica), so there's nothing 

374 # meaningful to be returned as the `raw` response, 

375 # hence we return `None`. 

376 return None 

377 return from_timezone_name(timezone_id, raw=response)

379 def _normalize_timezone_at_time(self, at_time): 

380 if at_time is None: 

381 timestamp = timegm(datetime.utcnow().utctimetuple()) 

382 elif isinstance(at_time, datetime): 

383 # Naive datetimes are silently treated as UTC. 

384 # Timezone-aware datetimes are handled correctly. 

385 timestamp = timegm(at_time.utctimetuple()) 

386 else: 

387 raise GeocoderQueryError( 

388 "`at_time` must be an instance of `datetime.datetime`" 

389 ) 

390 return timestamp

400 def parse_place(place): 

401 '''Get the location, lat, lng from a single json place.''' 

402 location = place.get('formatted_address') 

403 latitude = place['geometry']['location']['lat'] 

404 longitude = place['geometry']['location']['lng'] 

405 return Location(location, (latitude, longitude), place) 

406 

407 if exactly_one:

409 else:

412 def _check_status(self, status): 

413 # https://developers.google.com/maps/documentation/geocoding/overview#StatusCodes 

414 if status == 'ZERO_RESULTS': 

415 # When there are no results, just return. 

416 return 

417 if status in ('OVER_QUERY_LIMIT', 'OVER_DAILY_LIMIT'): 

418 raise GeocoderQuotaExceeded( 

419 'The given key has gone over the requests limit in the 24' 

420 ' hour period or has submitted too many requests in too' 

421 ' short a period of time.' 

422 ) 

423 elif status == 'REQUEST_DENIED': 

424 raise GeocoderQueryError( 

425 'Your request was denied.' 

426 ) 

427 elif status == 'INVALID_REQUEST': 

428 raise GeocoderQueryError('Probably missing address or latlng.') 

429 else: 

430 raise GeocoderQueryError('Unknown error.')

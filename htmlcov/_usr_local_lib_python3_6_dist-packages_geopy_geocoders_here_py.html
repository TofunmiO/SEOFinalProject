161 

162 This implementation supports only a subset of all available parameters.

167

170 `street`, `housenumber`, or `postalcode`. 

171 :type query: str or dict 

172 

173 :param bbox: A type of spatial filter, limits the search for any other attributes 

174 in the request. Specified by two coordinate (lat/lon) 

175 pairs -- corners of the box. `The bbox search is currently similar 

176 to mapview but it is not extended` (cited from the REST API docs). 

177 Relevant global results are also returned. 

178 Example: ``[Point(22, 180), Point(-22, -180)]``. 

179 :type bbox: list or tuple of 2 items of :class:`geopy.point.Point` or 

180 ``(latitude, longitude)`` or ``"%(latitude)s, %(longitude)s"``. 

181

183 by two lat/lon pairs -- corners of the bounding box, 

184 respectively. Matches from within the set map view plus an extended area 

185 are ranked highest. Relevant global results are also returned. 

186 Example: ``[Point(22, 180), Point(-22, -180)]``. 

187 :type mapview: list or tuple of 2 items of :class:`geopy.point.Point` or 

188 ``(latitude, longitude)`` or ``"%(latitude)s, %(longitude)s"``. 

189

191 available. 

192

195 the HERE API will return 10 results by default. This will be reset 

196 to one if ``exactly_one`` is True. 

197

200 ``maxresults`` is also provided.

203 must be a RFC 4647 language code, e.g. 'en-US'. 

204

206 https://developer.here.com/documentation/geocoder/topics/resource-params-additional.html. 

207 These will be added as one query parameter to the URL. 

208 

209 :param int timeout: Time, in seconds, to wait for the geocoding service

226 if bbox:

332 def _parse_json(self, doc, exactly_one=True): 

333 """ 

334 Parse a location name, latitude, and longitude from an JSON response. 

335 """ 

336 status_code = doc.get("statusCode", 200) 

337 if status_code != 200: 

338 err = doc.get("errorDetails", "") 

339 if status_code == 401: 

340 raise GeocoderAuthenticationFailure(err) 

341 elif status_code == 403: 

342 raise GeocoderInsufficientPrivileges(err) 

343 elif status_code == 429: 

344 raise GeocoderRateLimited(err) 

345 elif status_code == 503: 

346 raise GeocoderUnavailable(err) 

347 else: 

348 raise GeocoderServiceError(err) 

349 

350 try: 

351 resources = doc['Response']['View'][0]['Result'] 

352 except IndexError: 

353 resources = None 

354 if not resources: 

355 return None

357 def parse_resource(resource): 

358 """ 

359 Parse each return object. 

360 """ 

361 stripchars = ", \n" 

362 addr = resource['Location']['Address'] 

363 

364 address = addr.get('Label', '').strip(stripchars) 

365 city = addr.get('City', '').strip(stripchars) 

366 state = addr.get('State', '').strip(stripchars) 

367 zipcode = addr.get('PostalCode', '').strip(stripchars) 

368 country = addr.get('Country', '').strip(stripchars) 

369 

370 city_state = join_filter(", ", [city, state]) 

371 place = join_filter(" ", [city_state, zipcode]) 

372 location = join_filter(", ", [address, place, country]) 

373 

374 display_pos = resource['Location']['DisplayPosition'] 

375 latitude = float(display_pos['Latitude']) 

376 longitude = float(display_pos['Longitude']) 

377 

378 return Location(location, (latitude, longitude), resource) 

379 

380 if exactly_one: 

381 return parse_resource(resources[0]) 

382 else: 

383 return [parse_resource(resource) for resource in resources]

465 def geocode( 

466 self, 

467 query=None, 

468 *, 

469 components=None, 

470 at=None, 

471 countries=None, 

472 language=None, 

473 limit=None, 

474 exactly_one=True, 

475 timeout=DEFAULT_SENTINEL 

476 ): 

477 """ 

478 Return a location point by address. 

479 

480 :param str query: The address or query you wish to geocode. Optional, 

481 if ``components`` param is set. 

482 

483 :param dict components: A structured query. Can be used along with 

484 the free-text ``query``. Should be a dictionary whose keys 

485 are one of: 

486 `country`, `state`, `county`, `city`, `district`, `street`, 

487 `houseNumber`, `postalCode`. 

488 

489 :param at: The center of the search context. 

490 :type at: :class:`geopy.point.Point`, list or tuple of ``(latitude, 

491 longitude)``, or string as ``"%(latitude)s, %(longitude)s"``. 

492 

493 :param list countries: A list of country codes specified in 

494 `ISO 3166-1 alpha-3 <https://en.wikipedia.org/wiki/ISO_3166-1_alpha-3>`_ 

495 format, e.g. ``['USA', 'CAN']``. 

496 This is a hard filter. 

497 

498 :param str language: Affects the language of the response, 

499 must be a BCP 47 compliant language code, e.g. ``en-US``. 

500 

501 :param int limit: Defines the maximum number of items in the 

502 response structure. If not provided and there are multiple results 

503 the HERE API will return 20 results by default. This will be reset 

504 to one if ``exactly_one`` is True. 

505 

506 :param bool exactly_one: Return one result or a list of results, if 

507 available. 

508 

509 :param int timeout: Time, in seconds, to wait for the geocoding service 

510 to respond before raising a :class:`geopy.exc.GeocoderTimedOut` 

511 exception. Set this only if you wish to override, on this call 

512 only, the value set during the geocoder's initialization. 

513 

514 :rtype: ``None``, :class:`geopy.location.Location` or a list of them, if 

515 ``exactly_one=False``. 

516 """ 

517 params = { 

518 'apiKey': self.apikey, 

519 } 

520 

521 if query: 

522 params['q'] = query 

523 

524 if components: 

525 parts = [ 

526 "{}={}".format(key, val) 

527 for key, val 

528 in components.items() 

529 if key in self.structured_query_params 

530 ] 

531 if not parts: 

532 raise GeocoderQueryError("`components` dict must not be empty") 

533 for pair in parts: 

534 if ';' in pair: 

535 raise GeocoderQueryError( 

536 "';' must not be used in values of the structured query. " 

537 "Offending pair: {!r}".format(pair) 

538 ) 

539 params['qq'] = ';'.join(parts) 

540 

541 if at: 

542 point = self._coerce_point_to_string(at, output_format="%(lat)s,%(lon)s") 

543 params['at'] = point 

544 

545 if countries: 

546 params['in'] = 'countryCode:' + ','.join(countries) 

547 

548 if language: 

549 params['lang'] = language 

550 

551 if limit: 

552 params['limit'] = limit 

553 if exactly_one: 

554 params['limit'] = 1 

555 

556 url = "?".join((self.api, urlencode(params))) 

557 logger.debug("%s.geocode: %s", self.__class__.__name__, url) 

558 callback = partial(self._parse_json, exactly_one=exactly_one) 

559 return self._call_geocoder(url, callback, timeout=timeout)

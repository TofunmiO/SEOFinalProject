71def _ensure_data( 

72 values, dtype: Optional[DtypeObj] = None 

73) -> Tuple[np.ndarray, DtypeObj]: 

74 """ 

75 routine to ensure that our data is of the correct 

76 input dtype for lower-level routines 

77 

78 This will coerce: 

79 - ints -> int64 

80 - uint -> uint64 

81 - bool -> uint64 (TODO this should be uint8) 

82 - datetimelike -> i8 

83 - datetime64tz -> i8 (in local tz) 

84 - categorical -> codes 

85 

86 Parameters 

87 ---------- 

88 values : array-like 

89 dtype : pandas_dtype, optional

91 

92 Returns 

93 ------- 

94 values : ndarray 

95 pandas_dtype : np.dtype or ExtensionDtype 

96 """ 

97 

98 if not isinstance(values, ABCMultiIndex): 

99 # extract_array would raise 

100 values = extract_array(values, extract_numpy=True) 

101 

102 # we check some simple dtypes first 

103 if is_object_dtype(dtype): 

104 return ensure_object(np.asarray(values)), np.dtype("object") 

105 elif is_object_dtype(values) and dtype is None: 

106 return ensure_object(np.asarray(values)), np.dtype("object") 

107 

108 try: 

109 if is_bool_dtype(values) or is_bool_dtype(dtype): 

110 # we are actually coercing to uint64 

111 # until our algos support uint8 directly (see TODO) 

112 return np.asarray(values).astype("uint64"), np.dtype("bool") 

113 elif is_signed_integer_dtype(values) or is_signed_integer_dtype(dtype): 

114 return ensure_int64(values), np.dtype("int64") 

115 elif is_unsigned_integer_dtype(values) or is_unsigned_integer_dtype(dtype): 

116 return ensure_uint64(values), np.dtype("uint64") 

117 elif is_float_dtype(values) or is_float_dtype(dtype): 

118 return ensure_float64(values), np.dtype("float64") 

119 elif is_complex_dtype(values) or is_complex_dtype(dtype): 

120 

121 # ignore the fact that we are casting to float 

122 # which discards complex parts 

123 with catch_warnings(): 

124 simplefilter("ignore", np.ComplexWarning) 

125 values = ensure_float64(values) 

126 return values, np.dtype("float64") 

127 

128 except (TypeError, ValueError, OverflowError): 

129 # if we are trying to coerce to a dtype 

130 # and it is incompatible this will fall through to here 

131 return ensure_object(values), np.dtype("object") 

132 

133 # datetimelike 

134 vals_dtype = getattr(values, "dtype", None) 

135 if needs_i8_conversion(vals_dtype) or needs_i8_conversion(dtype): 

136 if is_period_dtype(vals_dtype) or is_period_dtype(dtype): 

137 from pandas import PeriodIndex

140 dtype = values.dtype 

141 elif is_timedelta64_dtype(vals_dtype) or is_timedelta64_dtype(dtype):

144 values = TimedeltaIndex(values)

146 else:

150 # Note: this is reached by DataFrame.rank calls GH#27027

152 asi8 = values.view("i8") 

153 dtype = values.dtype 

154 return asi8, dtype

157

159 dtype = values.dtype 

160 

161 return values.asi8, dtype 

162 

163 elif is_categorical_dtype(vals_dtype) and ( 

164 is_categorical_dtype(dtype) or dtype is None 

165 ): 

166 values = values.codes 

167 dtype = pandas_dtype("category") 

168 

169 # we are actually coercing to int64 

170 # until our algos support int* directly (not all do) 

171 values = ensure_int64(values) 

172

175 # we have failed, return object

311def unique(values): 

312 """ 

313 Hash table-based unique. Uniques are returned in order 

314 of appearance. This does NOT sort. 

315 

316 Significantly faster than numpy.unique. Includes NA values. 

317 

318 Parameters 

319 ---------- 

320 values : 1d array-like 

321 

322 Returns 

323 ------- 

324 numpy.ndarray or ExtensionArray 

325 

326 The return can be: 

327 

328 * Index : when the input is an Index 

329 * Categorical : when the input is a Categorical dtype 

330 * ndarray : when the input is a Series/ndarray 

331 

332 Return numpy.ndarray or ExtensionArray. 

333 

334 See Also 

335 -------- 

336 Index.unique : Return unique values from an Index. 

337 Series.unique : Return unique values of Series object. 

338 

339 Examples 

340 -------- 

341 >>> pd.unique(pd.Series([2, 1, 3, 3])) 

342 array([2, 1, 3]) 

343 

344 >>> pd.unique(pd.Series([2] + [1] * 5)) 

345 array([2, 1]) 

346 

347 >>> pd.unique(pd.Series([pd.Timestamp('20160101'), 

348 ... pd.Timestamp('20160101')])) 

349 array(['2016-01-01T00:00:00.000000000'], dtype='datetime64[ns]') 

350 

351 >>> pd.unique(pd.Series([pd.Timestamp('20160101', tz='US/Eastern'), 

352 ... pd.Timestamp('20160101', tz='US/Eastern')])) 

353 array([Timestamp('2016-01-01 00:00:00-0500', tz='US/Eastern')], 

354 dtype=object) 

355 

356 >>> pd.unique(pd.Index([pd.Timestamp('20160101', tz='US/Eastern'), 

357 ... pd.Timestamp('20160101', tz='US/Eastern')])) 

358 DatetimeIndex(['2016-01-01 00:00:00-05:00'], 

359 ... dtype='datetime64[ns, US/Eastern]', freq=None) 

360 

361 >>> pd.unique(list('baabc')) 

362 array(['b', 'a', 'c'], dtype=object) 

363 

364 An unordered Categorical will return categories in the 

365 order of appearance. 

366 

367 >>> pd.unique(pd.Series(pd.Categorical(list('baabc')))) 

368 [b, a, c] 

369 Categories (3, object): [b, a, c] 

370 

371 >>> pd.unique(pd.Series(pd.Categorical(list('baabc'), 

372 ... categories=list('abc')))) 

373 [b, a, c] 

374 Categories (3, object): [b, a, c] 

375 

376 An ordered Categorical preserves the category ordering. 

377 

378 >>> pd.unique(pd.Series(pd.Categorical(list('baabc'), 

379 ... categories=list('abc'), 

380 ... ordered=True))) 

381 [b, a, c] 

382 Categories (3, object): [a < b < c] 

383 

384 An array of tuples 

385 

386 >>> pd.unique([('a', 'b'), ('b', 'a'), ('a', 'c'), ('b', 'a')]) 

387 array([('a', 'b'), ('b', 'a'), ('a', 'c')], dtype=object) 

388 """ 

389 values = _ensure_arraylike(values) 

390 

391 if is_extension_array_dtype(values): 

392 # Dispatch to extension dtype's unique. 

393 return values.unique() 

394 

395 original = values 

396 htable, values = _get_hashtable_algo(values) 

397 

398 table = htable(len(values)) 

399 uniques = table.unique(values) 

400 uniques = _reconstruct_data(uniques, original.dtype, original) 

401 return uniques

422 raise TypeError(

983def checked_add_with_arr(arr, b, arr_mask=None, b_mask=None): 

984 """ 

985 Perform array addition that checks for underflow and overflow. 

986 

987 Performs the addition of an int64 array and an int64 integer (or array) 

988 but checks that they do not result in overflow first. For elements that 

989 are indicated to be NaN, whether or not there is overflow for that element 

990 is automatically ignored. 

991 

992 Parameters 

993 ---------- 

994 arr : array addend. 

995 b : array or scalar addend. 

996 arr_mask : boolean array or None 

997 array indicating which elements to exclude from checking 

998 b_mask : boolean array or boolean or None 

999 array or scalar indicating which element(s) to exclude from checking 

1000 

1001 Returns 

1002 ------- 

1003 sum : An array for elements x + b for each element x in arr if b is 

1004 a scalar or an array for elements x + y for each element pair 

1005 (x, y) in (arr, b). 

1006 

1007 Raises 

1008 ------ 

1009 OverflowError if any x + y exceeds the maximum or minimum int64 value. 

1010 """ 

1011 # For performance reasons, we broadcast 'b' to the new array 'b2' 

1012 # so that it has the same size as 'arr'. 

1013 b2 = np.broadcast_to(b, arr.shape) 

1014 if b_mask is not None:

1016 b2_mask = np.broadcast_to(b_mask, arr.shape) 

1017 else: 

1018 b2_mask = None 

1019

1021 # ignore whether they overflow or not when doing the checked add. 

1022 if arr_mask is not None and b2_mask is not None: 

1023 not_nan = np.logical_not(arr_mask | b2_mask) 

1024 elif arr_mask is not None: 

1025 not_nan = np.logical_not(arr_mask) 

1026 elif b_mask is not None: 

1027 not_nan = np.logical_not(b2_mask) 

1028 else: 

1029 not_nan = np.empty(arr.shape, dtype=bool) 

1030 not_nan.fill(True) 

1031 

1032 # gh-14324: For each element in 'arr' and its corresponding element 

1033 # in 'b2', we check the sign of the element in 'b2'. If it is positive, 

1034 # we then check whether its sum with the element in 'arr' exceeds 

1035 # np.iinfo(np.int64).max. If so, we have an overflow error. If it 

1036 # it is negative, we then check whether its sum with the element in 

1037 # 'arr' exceeds np.iinfo(np.int64).min. If so, we have an overflow 

1038 # error as well.

1040 mask2 = b2 < 0

1043 to_raise = ((np.iinfo(np.int64).min - b2 > arr) & not_nan).any()

1046 else:

1048 ((np.iinfo(np.int64).max - b2[mask1] < arr[mask1]) & not_nan[mask1]).any() 

1049 or ( 

1050 (np.iinfo(np.int64).min - b2[mask2] > arr[mask2]) & not_nan[mask2]

1052 )

1091 --------

1109 """

1547def take(arr, indices, axis: int = 0, allow_fill: bool = False, fill_value=None): 

1548 """

1550

1552

1559 Indices to be taken. 

1560 axis : int, default 0 

1561 The axis over which to select values. 

1562 allow_fill : bool, default False 

1563 How to handle negative values in `indices`. 

1564 

1565 * False: negative values in `indices` indicate positional indices 

1566 from the right (the default). This is similar to :func:`numpy.take`. 

1567 

1568 * True: negative values in `indices` indicate 

1569 missing values. These values are set to `fill_value`. Any other 

1570 other negative values raise a ``ValueError``. 

1571 

1572 fill_value : any, optional 

1573 Fill value to use for NA-indices when `allow_fill` is True. 

1574 This may be ``None``, in which case the default NA value for 

1575 the type (``self.dtype.na_value``) is used. 

1576 

1577 For multi-dimensional `arr`, each *element* is filled with 

1578 `fill_value`. 

1579 

1580 Returns 

1581 ------- 

1582 ndarray or ExtensionArray 

1583 Same type as the input. 

1584 

1585 Raises 

1586 ------ 

1587 IndexError 

1588 When `indices` is out of bounds for the array. 

1589 ValueError 

1590 When the indexer contains negative values other than ``-1`` 

1591 and `allow_fill` is True. 

1592 

1593 Notes 

1594 ----- 

1595 When `allow_fill` is False, `indices` may be whatever dimensionality 

1596 is accepted by NumPy for `arr`. 

1597 

1598 When `allow_fill` is True, `indices` should be 1-D. 

1599

1602 numpy.take : Take elements from an array along an axis. 

1603 

1604 Examples 

1605 -------- 

1606 >>> from pandas.api.extensions import take

1609 positional indices from the right.

1613 

1614 Setting ``allow_fill=True`` will place `fill_value` in those positions. 

1615 

1616 >>> take(np.array([10, 20, 30]), [0, 0, -1], allow_fill=True) 

1617 array([10., 10., nan]) 

1618 

1619 >>> take(np.array([10, 20, 30]), [0, 0, -1], allow_fill=True,

1807def searchsorted(arr, value, side="left", sorter=None): 

1808 """ 

1809 Find indices where elements should be inserted to maintain order. 

1810 

1811 .. versionadded:: 0.25.0 

1812 

1813 Find the indices into a sorted array `arr` (a) such that, if the 

1814 corresponding elements in `value` were inserted before the indices, 

1815 the order of `arr` would be preserved. 

1816 

1817 Assuming that `arr` is sorted: 

1818 

1819 ====== ================================ 

1820 `side` returned index `i` satisfies 

1821 ====== ================================ 

1822 left ``arr[i-1] < value <= self[i]`` 

1823 right ``arr[i-1] <= value < self[i]`` 

1824 ====== ================================ 

1825 

1826 Parameters 

1827 ---------- 

1828 arr: array-like 

1829 Input array. If `sorter` is None, then it must be sorted in 

1830 ascending order, otherwise `sorter` must be an array of indices 

1831 that sort it. 

1832 value : array_like 

1833 Values to insert into `arr`. 

1834 side : {'left', 'right'}, optional 

1835 If 'left', the index of the first suitable location found is given. 

1836 If 'right', return the last such index. If there is no suitable 

1837 index, return either 0 or N (where N is the length of `self`). 

1838 sorter : 1-D array_like, optional 

1839 Optional array of integer indices that sort array a into ascending 

1840 order. They are typically the result of argsort. 

1841 

1842 Returns 

1843 ------- 

1844 array of ints 

1845 Array of insertion points with the same shape as `value`. 

1846 

1847 See Also 

1848 -------- 

1849 numpy.searchsorted : Similar method from NumPy. 

1850 """ 

1851 if sorter is not None: 

1852 sorter = ensure_platform_int(sorter) 

1853 

1854 if ( 

1855 isinstance(arr, np.ndarray) 

1856 and is_integer_dtype(arr) 

1857 and (is_integer(value) or is_integer_dtype(value)) 

1858 ): 

1859 # if `arr` and `value` have different dtypes, `arr` would be 

1860 # recast by numpy, causing a slow search. 

1861 # Before searching below, we therefore try to give `value` the 

1862 # same dtype as `arr`, while guarding against integer overflows. 

1863 iinfo = np.iinfo(arr.dtype.type) 

1864 value_arr = np.array([value]) if is_scalar(value) else np.array(value) 

1865 if (value_arr >= iinfo.min).all() and (value_arr <= iinfo.max).all(): 

1866 # value within bounds, so no overflow, so can convert value dtype 

1867 # to dtype of arr 

1868 dtype = arr.dtype 

1869 else: 

1870 dtype = value_arr.dtype 

1871 

1872 if is_scalar(value): 

1873 value = dtype.type(value) 

1874 else: 

1875 value = array(value, dtype=dtype) 

1876 elif not ( 

1877 is_object_dtype(arr) or is_numeric_dtype(arr) or is_categorical_dtype(arr) 

1878 ): 

1879 # E.g. if `arr` is an array with dtype='datetime64[ns]' 

1880 # and `value` is a pd.Timestamp, we may need to convert value 

1881 value_ser = array([value]) if is_scalar(value) else array(value) 

1882 value = value_ser[0] if is_scalar(value) else value_ser 

1883 if isinstance(value, Timestamp) and value.tzinfo is None: 

1884 value = value.to_datetime64() 

1885 

1886 result = arr.searchsorted(value, side=side, sorter=sorter) 

1887 return result

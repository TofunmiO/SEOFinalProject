31def concatenate_block_managers( 

32 mgrs_indexers, axes, concat_axis: int, copy: bool, 

33) -> BlockManager: 

34 """ 

35 Concatenate block managers into one. 

36 

37 Parameters 

38 ---------- 

39 mgrs_indexers : list of (BlockManager, {axis: indexer,...}) tuples 

40 axes : list of Index 

41 concat_axis : int 

42 copy : bool 

43 

44 Returns 

45 ------- 

46 BlockManager 

47 """ 

48 concat_plans = [ 

49 _get_mgr_concatenation_plan(mgr, indexers) for mgr, indexers in mgrs_indexers 

50 ] 

51 concat_plan = _combine_concat_plans(concat_plans, concat_axis) 

52 blocks = [] 

53 

54 for placement, join_units in concat_plan: 

55 

56 if len(join_units) == 1 and not join_units[0].indexers: 

57 b = join_units[0].block 

58 values = b.values 

59 if copy: 

60 values = values.copy() 

61 else: 

62 values = values.view() 

63 b = b.make_block_same_class(values, placement=placement) 

64 elif _is_uniform_join_units(join_units): 

65 blk = join_units[0].block 

66 vals = [ju.block.values for ju in join_units] 

67 

68 if not blk.is_extension: 

69 values = concat_compat(vals, axis=blk.ndim - 1) 

70 else: 

71 # TODO(EA2D): special-casing not needed with 2D EAs 

72 values = concat_compat(vals) 

73 if not isinstance(values, ExtensionArray): 

74 values = values.reshape(1, len(values)) 

75 

76 b = make_block(values, placement=placement, ndim=blk.ndim) 

77 else: 

78 b = make_block( 

79 _concatenate_join_units(join_units, concat_axis, copy=copy,), 

80 placement=placement, 

81 ) 

82 blocks.append(b) 

83 

84 return BlockManager(blocks, axes)

87def _get_mgr_concatenation_plan(mgr, indexers): 

88 """ 

89 Construct concatenation plan for given block manager and indexers. 

90 

91 Parameters 

92 ---------- 

93 mgr : BlockManager 

94 indexers : dict of {axis: indexer} 

95 

96 Returns 

97 ------- 

98 plan : list of (BlockPlacement, JoinUnit) tuples 

99 

100 """ 

101 # Calculate post-reindex shape , save for item axis which will be separate 

102 # for each block anyway. 

103 mgr_shape = list(mgr.shape) 

104 for ax, indexer in indexers.items(): 

105 mgr_shape[ax] = len(indexer) 

106 mgr_shape = tuple(mgr_shape) 

107 

108 if 0 in indexers: 

109 ax0_indexer = indexers.pop(0) 

110 blknos = algos.take_1d(mgr.blknos, ax0_indexer, fill_value=-1) 

111 blklocs = algos.take_1d(mgr.blklocs, ax0_indexer, fill_value=-1) 

112 else: 

113 

114 if mgr._is_single_block: 

115 blk = mgr.blocks[0] 

116 return [(blk.mgr_locs, JoinUnit(blk, mgr_shape, indexers))] 

117 

118 ax0_indexer = None 

119 blknos = mgr.blknos 

120 blklocs = mgr.blklocs 

121 

122 plan = [] 

123 for blkno, placements in libinternals.get_blkno_placements(blknos, group=False): 

124 

125 assert placements.is_slice_like 

126 

127 join_unit_indexers = indexers.copy() 

128 

129 shape = list(mgr_shape) 

130 shape[0] = len(placements) 

131 shape = tuple(shape) 

132 

133 if blkno == -1: 

134 unit = JoinUnit(None, shape) 

135 else: 

136 blk = mgr.blocks[blkno] 

137 ax0_blk_indexer = blklocs[placements.indexer] 

138 

139 unit_no_ax0_reindexing = ( 

140 len(placements) == len(blk.mgr_locs) 

141 and 

142 # Fastpath detection of join unit not 

143 # needing to reindex its block: no ax0 

144 # reindexing took place and block 

145 # placement was sequential before. 

146 ( 

147 ( 

148 ax0_indexer is None 

149 and blk.mgr_locs.is_slice_like 

150 and blk.mgr_locs.as_slice.step == 1 

151 ) 

152 or 

153 # Slow-ish detection: all indexer locs 

154 # are sequential (and length match is 

155 # checked above). 

156 (np.diff(ax0_blk_indexer) == 1).all() 

157 ) 

158 ) 

159 

160 # Omit indexer if no item reindexing is required. 

161 if unit_no_ax0_reindexing: 

162 join_unit_indexers.pop(0, None) 

163 else: 

164 join_unit_indexers[0] = ax0_blk_indexer 

165 

166 unit = JoinUnit(blk, shape, join_unit_indexers) 

167 

168 plan.append((placements, unit)) 

169 

170 return plan

174 def __init__(self, block, shape, indexers=None): 

175 # Passing shape explicitly is required for cases when block is None. 

176 if indexers is None: 

177 indexers = {} 

178 self.block = block 

179 self.indexers = indexers 

180 self.shape = shape

182 def __repr__(self) -> str: 

183 return f"{type(self).__name__}({repr(self.block)}, {self.indexers})"

185 @cache_readonly 

186 def needs_filling(self): 

187 for indexer in self.indexers.values(): 

188 # FIXME: cache results of indexer == -1 checks. 

189 if (indexer == -1).any(): 

190 return True 

191 

192 return False

194 @cache_readonly 

195 def dtype(self): 

196 if self.block is None: 

197 raise AssertionError("Block is None, no dtype") 

198 

199 if not self.needs_filling: 

200 return self.block.dtype 

201 else: 

202 return _get_dtype(maybe_promote(self.block.dtype, self.block.fill_value)[0]) 

203 

204 @cache_readonly 

205 def is_na(self): 

206 if self.block is None: 

207 return True 

208 

209 if not self.block._can_hold_na: 

210 return False 

211 

212 # Usually it's enough to check but a small fraction of values to see if 

213 # a block is NOT null, chunks should help in such cases. 1000 value 

214 # was chosen rather arbitrarily. 

215 values = self.block.values 

216 if self.block.is_categorical: 

217 values_flat = values.categories 

218 elif is_sparse(self.block.values.dtype): 

219 return False 

220 elif self.block.is_extension: 

221 # TODO(EA2D): no need for special case with 2D EAs 

222 values_flat = values 

223 else: 

224 values_flat = values.ravel(order="K") 

225 total_len = values_flat.shape[0] 

226 chunk_len = max(total_len // 40, 1000) 

227 for i in range(0, total_len, chunk_len): 

228 if not isna(values_flat[i : i + chunk_len]).all(): 

229 return False 

230 

231 return True

233 def get_reindexed_values(self, empty_dtype, upcasted_na): 

234 if upcasted_na is None: 

235 # No upcasting is necessary 

236 fill_value = self.block.fill_value 

237 values = self.block.get_values() 

238 else: 

239 fill_value = upcasted_na 

240 

241 if self.is_na: 

242 if getattr(self.block, "is_object", False): 

243 # we want to avoid filling with np.nan if we are 

244 # using None; we already know that we are all 

245 # nulls 

246 values = self.block.values.ravel(order="K") 

247 if len(values) and values[0] is None: 

248 fill_value = None 

249 

250 if getattr(self.block, "is_datetimetz", False) or is_datetime64tz_dtype( 

251 empty_dtype 

252 ): 

253 if self.block is None: 

254 array = empty_dtype.construct_array_type() 

255 # TODO(EA2D): special case unneeded with 2D EAs 

256 return array( 

257 np.full(self.shape[1], fill_value.value), dtype=empty_dtype 

258 ) 

259 elif getattr(self.block, "is_categorical", False): 

260 pass 

261 elif getattr(self.block, "is_extension", False): 

262 pass 

263 elif is_extension_array_dtype(empty_dtype): 

264 missing_arr = empty_dtype.construct_array_type()._from_sequence( 

265 [], dtype=empty_dtype 

266 ) 

267 ncols, nrows = self.shape 

268 assert ncols == 1, ncols 

269 empty_arr = -1 * np.ones((nrows,), dtype=np.intp) 

270 return missing_arr.take( 

271 empty_arr, allow_fill=True, fill_value=fill_value 

272 ) 

273 else: 

274 missing_arr = np.empty(self.shape, dtype=empty_dtype) 

275 missing_arr.fill(fill_value) 

276 return missing_arr 

277 

278 if (not self.indexers) and (not self.block._can_consolidate): 

279 # preserve these for validation in concat_compat 

280 return self.block.values 

281 

282 if self.block.is_bool and not self.block.is_categorical: 

283 # External code requested filling/upcasting, bool values must 

284 # be upcasted to object to avoid being upcasted to numeric. 

285 values = self.block.astype(np.object_).values 

286 elif self.block.is_extension: 

287 values = self.block.values 

288 else: 

289 # No dtype upcasting is done here, it will be performed during 

290 # concatenation itself. 

291 values = self.block.values 

292 

293 if not self.indexers: 

294 # If there's no indexing to be done, we want to signal outside 

295 # code that this array must be copied explicitly. This is done 

296 # by returning a view and checking `retval.base`. 

297 values = values.view() 

298 

299 else: 

300 for ax, indexer in self.indexers.items(): 

301 values = algos.take_nd(values, indexer, axis=ax, fill_value=fill_value) 

302 

303 return values

306def _concatenate_join_units(join_units, concat_axis, copy): 

307 """ 

308 Concatenate values from several join units along selected axis. 

309 """ 

310 if concat_axis == 0 and len(join_units) > 1: 

311 # Concatenating join units along ax0 is handled in _merge_blocks. 

312 raise AssertionError("Concatenating join units along axis0") 

313 

314 empty_dtype, upcasted_na = _get_empty_dtype_and_na(join_units) 

315 

316 to_concat = [ 

317 ju.get_reindexed_values(empty_dtype=empty_dtype, upcasted_na=upcasted_na) 

318 for ju in join_units 

319 ] 

320 

321 if len(to_concat) == 1: 

322 # Only one block, nothing to concatenate. 

323 concat_values = to_concat[0] 

324 if copy: 

325 if isinstance(concat_values, np.ndarray): 

326 # non-reindexed (=not yet copied) arrays are made into a view 

327 # in JoinUnit.get_reindexed_values 

328 if concat_values.base is not None: 

329 concat_values = concat_values.copy() 

330 else: 

331 concat_values = concat_values.copy() 

332 elif any(isinstance(t, ExtensionArray) for t in to_concat): 

333 # concatting with at least one EA means we are concatting a single column 

334 # the non-EA values are 2D arrays with shape (1, n) 

335 to_concat = [t if isinstance(t, ExtensionArray) else t[0, :] for t in to_concat] 

336 concat_values = concat_compat(to_concat, axis=0) 

337 if not isinstance(concat_values, ExtensionArray) or ( 

338 isinstance(concat_values, DatetimeArray) and concat_values.tz is None 

339 ): 

340 # if the result of concat is not an EA but an ndarray, reshape to 

341 # 2D to put it a non-EA Block 

342 # special case DatetimeArray, which *is* an EA, but is put in a 

343 # consolidated 2D block 

344 concat_values = np.atleast_2d(concat_values) 

345 else: 

346 concat_values = concat_compat(to_concat, axis=concat_axis,) 

347 

348 return concat_values

351def _get_empty_dtype_and_na(join_units): 

352 """ 

353 Return dtype and N/A values to use when concatenating specified units. 

354 

355 Returned N/A value may be None which means there was no casting involved. 

356 

357 Returns 

358 ------- 

359 dtype 

360 na 

361 """ 

362 if len(join_units) == 1: 

363 blk = join_units[0].block 

364 if blk is None: 

365 return np.dtype(np.float64), np.nan 

366 

367 if _is_uniform_reindex(join_units): 

368 # FIXME: integrate property 

369 empty_dtype = join_units[0].block.dtype 

370 upcasted_na = join_units[0].block.fill_value 

371 return empty_dtype, upcasted_na 

372 

373 has_none_blocks = False 

374 dtypes = [None] * len(join_units) 

375 for i, unit in enumerate(join_units): 

376 if unit.block is None: 

377 has_none_blocks = True 

378 else: 

379 dtypes[i] = unit.dtype 

380 

381 upcast_classes = defaultdict(list) 

382 null_upcast_classes = defaultdict(list) 

383 for dtype, unit in zip(dtypes, join_units): 

384 if dtype is None: 

385 continue 

386 

387 if is_categorical_dtype(dtype): 

388 upcast_cls = "category" 

389 elif is_datetime64tz_dtype(dtype): 

390 upcast_cls = "datetimetz" 

391 

392 elif is_extension_array_dtype(dtype): 

393 upcast_cls = "extension" 

394 

395 elif issubclass(dtype.type, np.bool_): 

396 upcast_cls = "bool" 

397 elif issubclass(dtype.type, np.object_): 

398 upcast_cls = "object" 

399 elif is_datetime64_dtype(dtype): 

400 upcast_cls = "datetime" 

401 elif is_timedelta64_dtype(dtype): 

402 upcast_cls = "timedelta" 

403 elif is_sparse(dtype): 

404 upcast_cls = dtype.subtype.name 

405 elif is_float_dtype(dtype) or is_numeric_dtype(dtype): 

406 upcast_cls = dtype.name 

407 else: 

408 upcast_cls = "float" 

409 

410 # Null blocks should not influence upcast class selection, unless there 

411 # are only null blocks, when same upcasting rules must be applied to 

412 # null upcast classes. 

413 if unit.is_na: 

414 null_upcast_classes[upcast_cls].append(dtype) 

415 else: 

416 upcast_classes[upcast_cls].append(dtype) 

417 

418 if not upcast_classes: 

419 upcast_classes = null_upcast_classes 

420 # TODO: de-duplicate with maybe_promote? 

421 # create the result 

422 if "extension" in upcast_classes: 

423 if len(upcast_classes) == 1: 

424 cls = upcast_classes["extension"][0] 

425 return cls, cls.na_value 

426 else: 

427 return np.dtype("object"), np.nan 

428 elif "object" in upcast_classes: 

429 return np.dtype(np.object_), np.nan 

430 elif "bool" in upcast_classes: 

431 if has_none_blocks: 

432 return np.dtype(np.object_), np.nan 

433 else: 

434 return np.dtype(np.bool_), None 

435 elif "category" in upcast_classes: 

436 return np.dtype(np.object_), np.nan 

437 elif "datetimetz" in upcast_classes: 

438 # GH-25014. We use NaT instead of iNaT, since this eventually 

439 # ends up in DatetimeArray.take, which does not allow iNaT. 

440 dtype = upcast_classes["datetimetz"] 

441 return dtype[0], NaT 

442 elif "datetime" in upcast_classes: 

443 return np.dtype("M8[ns]"), np.datetime64("NaT", "ns") 

444 elif "timedelta" in upcast_classes: 

445 return np.dtype("m8[ns]"), np.timedelta64("NaT", "ns") 

446 else: # pragma 

447 try: 

448 g = np.find_common_type(upcast_classes, []) 

449 except TypeError: 

450 # At least one is an ExtensionArray 

451 return np.dtype(np.object_), np.nan 

452 else: 

453 if is_float_dtype(g): 

454 return g, g.type(np.nan) 

455 elif is_numeric_dtype(g): 

456 if has_none_blocks: 

457 return np.dtype(np.float64), np.nan 

458 else: 

459 return g, None 

460 

461 msg = "invalid dtype determination in get_concat_dtype" 

462 raise AssertionError(msg)

465def _is_uniform_join_units(join_units: List[JoinUnit]) -> bool: 

466 """ 

467 Check if the join units consist of blocks of uniform type that can 

468 be concatenated using Block.concat_same_type instead of the generic 

469 _concatenate_join_units (which uses `concat_compat`). 

470 

471 """ 

472 # TODO: require dtype match in addition to same type? e.g. DatetimeTZBlock 

473 # cannot necessarily join 

474 return ( 

475 # all blocks need to have the same type 

476 all(type(ju.block) is type(join_units[0].block) for ju in join_units) 

477 and # noqa 

478 # no blocks that would get missing values (can lead to type upcasts) 

479 # unless we're an extension dtype. 

480 all(not ju.is_na or ju.block.is_extension for ju in join_units)

482 # no blocks with indexers (as then the dimensions do not fit) 

483 all(not ju.indexers for ju in join_units) 

484 and 

485 # only use this path when there is something to concatenate 

486 len(join_units) > 1 

487 )

498def _trim_join_unit(join_unit, length): 

499 """ 

500 Reduce join_unit's shape along item axis to length. 

501 

502 Extra items that didn't fit are returned as a separate block. 

503 """ 

504 if 0 not in join_unit.indexers: 

505 extra_indexers = join_unit.indexers 

506 

507 if join_unit.block is None: 

508 extra_block = None 

509 else: 

510 extra_block = join_unit.block.getitem_block(slice(length, None)) 

511 join_unit.block = join_unit.block.getitem_block(slice(length)) 

512 else: 

513 extra_block = join_unit.block 

514 

515 extra_indexers = copy.copy(join_unit.indexers) 

516 extra_indexers[0] = extra_indexers[0][length:] 

517 join_unit.indexers[0] = join_unit.indexers[0][:length] 

518 

519 extra_shape = (join_unit.shape[0] - length,) + join_unit.shape[1:] 

520 join_unit.shape = (length,) + join_unit.shape[1:] 

521 

522 return JoinUnit(block=extra_block, indexers=extra_indexers, shape=extra_shape)

525def _combine_concat_plans(plans, concat_axis): 

526 """ 

527 Combine multiple concatenation plans into one. 

528 

529 existing_plan is updated in-place. 

530 """ 

531 if len(plans) == 1: 

532 for p in plans[0]: 

533 yield p[0], [p[1]] 

534 

535 elif concat_axis == 0: 

536 offset = 0 

537 for plan in plans: 

538 last_plc = None 

539 

540 for plc, unit in plan: 

541 yield plc.add(offset), [unit] 

542 last_plc = plc 

543 

544 if last_plc is not None: 

545 offset += last_plc.as_slice.stop 

546 

547 else: 

548 num_ended = [0] 

549 

550 def _next_or_none(seq): 

551 retval = next(seq, None) 

552 if retval is None: 

553 num_ended[0] += 1 

554 return retval 

555 

556 plans = list(map(iter, plans)) 

557 next_items = list(map(_next_or_none, plans)) 

558 

559 while num_ended[0] != len(next_items): 

560 if num_ended[0] > 0: 

561 raise ValueError("Plan shapes are not aligned") 

562 

563 placements, units = zip(*next_items) 

564 

565 lengths = list(map(len, placements)) 

566 min_len, max_len = min(lengths), max(lengths) 

567 

568 if min_len == max_len: 

569 yield placements[0], units 

570 next_items[:] = map(_next_or_none, plans) 

571 else: 

572 yielded_placement = None 

573 yielded_units = [None] * len(next_items) 

574 for i, (plc, unit) in enumerate(next_items): 

575 yielded_units[i] = unit 

576 if len(plc) > min_len: 

577 # _trim_join_unit updates unit in place, so only 

578 # placement needs to be sliced to skip min_len. 

579 next_items[i] = (plc[min_len:], _trim_join_unit(unit, min_len)) 

580 else: 

581 yielded_placement = plc 

582 next_items[i] = _next_or_none(plans[i]) 

583 

584 yield yielded_placement, yielded_units

83def jit_user_function( 

84 func: Callable, nopython: bool, nogil: bool, parallel: bool 

85) -> Callable: 

86 """ 

87 JIT the user's function given the configurable arguments. 

88 

89 Parameters 

90 ---------- 

91 func : function 

92 user defined function 

93 nopython : bool 

94 nopython parameter for numba.JIT 

95 nogil : bool 

96 nogil parameter for numba.JIT 

97 parallel : bool 

98 parallel parameter for numba.JIT 

99 

100 Returns 

101 ------- 

102 function 

103 Numba JITed function 

104 """ 

105 numba = import_optional_dependency("numba") 

106 

107 if LooseVersion(numba.__version__) >= LooseVersion("0.49.0"): 

108 is_jitted = numba.extending.is_jitted(func) 

109 else: 

110 is_jitted = isinstance(func, numba.targets.registry.CPUDispatcher) 

111 

112 if is_jitted: 

113 # Don't jit a user passed jitted function 

114 numba_func = func 

115 else: 

116 

117 @numba.generated_jit(nopython=nopython, nogil=nogil, parallel=parallel) 

118 def numba_func(data, *_args): 

119 if getattr(np, func.__name__, False) is func or isinstance( 

120 func, types.BuiltinFunctionType 

121 ): 

122 jf = func 

123 else: 

124 jf = numba.jit(func, nopython=nopython, nogil=nogil) 

125 

126 def impl(data, *_args): 

127 return jf(data, *_args) 

128 

129 return impl 

130 

131 return numba_func

150def validate_udf(func: Callable) -> None: 

151 """ 

152 Validate user defined function for ops when using Numba. 

153 

154 The first signature arguments should include: 

155 

156 def f(values, index, ...): 

157 ... 

158 

159 Parameters 

160 ---------- 

161 func : function, default False 

162 user defined function 

163 

164 Returns 

165 ------- 

166 None 

167 

168 Raises 

169 ------ 

170 NumbaUtilError 

171 """ 

172 udf_signature = list(inspect.signature(func).parameters.keys()) 

173 expected_args = ["values", "index"] 

174 min_number_args = len(expected_args) 

175 if ( 

176 len(udf_signature) < min_number_args 

177 or udf_signature[:min_number_args] != expected_args 

178 ): 

179 raise NumbaUtilError( 

180 f"The first {min_number_args} arguments to {func.__name__} must be " 

181 f"{expected_args}" 

182 )

185def generate_numba_func( 

186 func: Callable, 

187 engine_kwargs: Optional[Dict[str, bool]], 

188 kwargs: dict, 

189 cache_key_str: str, 

190) -> Tuple[Callable, Tuple[Callable, str]]: 

191 """ 

192 Return a JITed function and cache key for the NUMBA_FUNC_CACHE 

193 

194 This _may_ be specific to groupby (as it's only used there currently). 

195 

196 Parameters 

197 ---------- 

198 func : function 

199 user defined function 

200 engine_kwargs : dict or None 

201 numba.jit arguments 

202 kwargs : dict 

203 kwargs for func 

204 cache_key_str : str 

205 string representing the second part of the cache key tuple 

206 

207 Returns 

208 ------- 

209 (JITed function, cache key) 

210 

211 Raises 

212 ------ 

213 NumbaUtilError 

214 """ 

215 nopython, nogil, parallel = get_jit_arguments(engine_kwargs) 

216 check_kwargs_and_nopython(kwargs, nopython) 

217 validate_udf(func) 

218 cache_key = (func, cache_key_str) 

219 numba_func = NUMBA_FUNC_CACHE.get( 

220 cache_key, jit_user_function(func, nopython, nogil, parallel) 

221 ) 

222 return numba_func, cache_key

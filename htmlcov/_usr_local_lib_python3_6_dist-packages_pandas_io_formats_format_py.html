1505def format_percentiles( 

1506 percentiles: Union[ 

1507 np.ndarray, List[Union[int, float]], List[float], List[Union[str, float]] 

1508 ] 

1509) -> List[str]: 

1510 """ 

1511 Outputs rounded and formatted percentiles. 

1512 

1513 Parameters 

1514 ---------- 

1515 percentiles : list-like, containing floats from interval [0,1] 

1516 

1517 Returns 

1518 ------- 

1519 formatted : list of strings 

1520 

1521 Notes 

1522 ----- 

1523 Rounding precision is chosen so that: (1) if any two elements of 

1524 ``percentiles`` differ, they remain different after rounding 

1525 (2) no entry is *rounded* to 0% or 100%. 

1526 Any non-integer is always rounded to at least 1 decimal place. 

1527 

1528 Examples 

1529 -------- 

1530 Keeps all entries different after rounding: 

1531 

1532 >>> format_percentiles([0.01999, 0.02001, 0.5, 0.666666, 0.9999]) 

1533 ['1.999%', '2.001%', '50%', '66.667%', '99.99%'] 

1534 

1535 No element is rounded to 0% or 100% (unless already equal to it). 

1536 Duplicates are allowed: 

1537 

1538 >>> format_percentiles([0, 0.5, 0.02001, 0.5, 0.666666, 0.9999]) 

1539 ['0%', '50%', '2.0%', '50%', '66.67%', '99.99%'] 

1540 """ 

1541 percentiles = np.asarray(percentiles) 

1542 

1543 # It checks for np.NaN as well 

1544 with np.errstate(invalid="ignore"): 

1545 if ( 

1546 not is_numeric_dtype(percentiles) 

1547 or not np.all(percentiles >= 0) 

1548 or not np.all(percentiles <= 1) 

1549 ): 

1550 raise ValueError("percentiles should all be in the interval [0,1]") 

1551 

1552 percentiles = 100 * percentiles 

1553 int_idx = np.isclose(percentiles.astype(int), percentiles) 

1554 

1555 if np.all(int_idx): 

1556 out = percentiles.astype(int).astype(str) 

1557 return [i + "%" for i in out] 

1558 

1559 unique_pcts = np.unique(percentiles) 

1560 to_begin = unique_pcts[0] if unique_pcts[0] > 0 else None 

1561 to_end = 100 - unique_pcts[-1] if unique_pcts[-1] < 100 else None 

1562 

1563 # Least precision that keeps percentiles unique after rounding 

1564 prec = -np.floor( 

1565 np.log10(np.min(np.ediff1d(unique_pcts, to_begin=to_begin, to_end=to_end))) 

1566 ).astype(int) 

1567 prec = max(1, prec) 

1568 out = np.empty_like(percentiles, dtype=object) 

1569 out[int_idx] = percentiles[int_idx].astype(int).astype(str) 

1570 out[~int_idx] = percentiles[~int_idx].round(prec).astype(str) 

1571 return [i + "%" for i in out]

1837 def __call__(self, num: Union[int, float]) -> str: 

1838 """ 

1839 Formats a number in engineering notation, appending a letter 

1840 representing the power of 1000 of the original number. Some examples: 

1841 

1842 >>> format_eng(0) # for self.accuracy = 0 

1843 ' 0' 

1844 

1845 >>> format_eng(1000000) # for self.accuracy = 1, 

1846 # self.use_eng_prefix = True 

1847 ' 1.0M' 

1848 

1849 >>> format_eng("-1e-6") # for self.accuracy = 2 

1850 # self.use_eng_prefix = False 

1851 '-1.00E-06' 

1852 

1853 @param num: the value to represent 

1854 @type num: either a numeric value or a string that can be converted to 

1855 a numeric value (as per decimal.Decimal constructor) 

1856 

1857 @return: engineering formatted string 

1858 """ 

1859 dnum = decimal.Decimal(str(num)) 

1860 

1861 if decimal.Decimal.is_nan(dnum): 

1862 return "NaN" 

1863 

1864 if decimal.Decimal.is_infinite(dnum): 

1865 return "inf" 

1866 

1867 sign = 1 

1868 

1869 if dnum < 0: # pragma: no cover 

1870 sign = -1 

1871 dnum = -dnum 

1872 

1873 if dnum != 0: 

1874 pow10 = decimal.Decimal(int(math.floor(dnum.log10() / 3) * 3)) 

1875 else: 

1876 pow10 = decimal.Decimal(0) 

1877 

1878 pow10 = pow10.min(max(self.ENG_PREFIXES.keys())) 

1879 pow10 = pow10.max(min(self.ENG_PREFIXES.keys())) 

1880 int_pow10 = int(pow10) 

1881 

1882 if self.use_eng_prefix: 

1883 prefix = self.ENG_PREFIXES[int_pow10] 

1884 else: 

1885 if int_pow10 < 0: 

1886 prefix = f"E-{-int_pow10:02d}" 

1887 else: 

1888 prefix = f"E+{int_pow10:02d}" 

1889 

1890 mant = sign * dnum / (10 ** pow10) 

1891 

1892 if self.accuracy is None: # pragma: no cover 

1893 format_str = "{mant: g}{prefix}" 

1894 else: 

1895 format_str = f"{{mant: .{self.accuracy:d}f}}{{prefix}}" 

1896 

1897 formatted = format_str.format(mant=mant, prefix=prefix) 

1898

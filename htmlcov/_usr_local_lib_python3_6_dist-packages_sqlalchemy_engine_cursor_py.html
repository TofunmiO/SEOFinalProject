109 def _adapt_to_context(self, context): 

110 """When using a cached Compiled construct that has a _result_map, 

111 for a new statement that used the cached Compiled, we need to ensure 

112 the keymap has the Column objects from our new statement as keys. 

113 So here we rewrite keymap with new entries for the new columns 

114 as matched to those of the cached statement. 

115 

116 """ 

117 

118 if not context.compiled._result_columns: 

119 return self 

120 

121 compiled_statement = context.compiled.statement 

122 invoked_statement = context.invoked_statement 

123 

124 if compiled_statement is invoked_statement: 

125 return self 

126 

127 # make a copy and add the columns from the invoked statement 

128 # to the result map. 

129 md = self.__class__.__new__(self.__class__) 

130 

131 md._keymap = dict(self._keymap) 

132 

133 keymap_by_position = self._keymap_by_result_column_idx 

134 

135 for idx, new in enumerate(invoked_statement._all_selected_columns): 

136 try: 

137 rec = keymap_by_position[idx] 

138 except KeyError: 

139 # this can happen when there are bogus column entries 

140 # in a TextualSelect 

141 pass 

142 else: 

143 md._keymap[new] = rec 

144 

145 md.case_sensitive = self.case_sensitive 

146 md._processors = self._processors 

147 assert not self._tuplefilter 

148 md._tuplefilter = None 

149 md._translated_indexes = None 

150 md._keys = self._keys 

151 md._keymap_by_result_column_idx = self._keymap_by_result_column_idx 

152 md._safe_for_cache = self._safe_for_cache 

153 return md

286 def _merge_cursor_description( 

287 self, 

288 context, 

289 cursor_description, 

290 result_columns, 

291 num_ctx_cols, 

292 cols_are_ordered, 

293 textual_ordered, 

294 loose_column_name_matching, 

295 ): 

296 """Merge a cursor.description with compiled result column information. 

297 

298 There are at least four separate strategies used here, selected 

299 depending on the type of SQL construct used to start with. 

300 

301 The most common case is that of the compiled SQL expression construct, 

302 which generated the column names present in the raw SQL string and 

303 which has the identical number of columns as were reported by 

304 cursor.description. In this case, we assume a 1-1 positional mapping 

305 between the entries in cursor.description and the compiled object. 

306 This is also the most performant case as we disregard extracting / 

307 decoding the column names present in cursor.description since we 

308 already have the desired name we generated in the compiled SQL 

309 construct. 

310 

311 The next common case is that of the completely raw string SQL, 

312 such as passed to connection.execute(). In this case we have no 

313 compiled construct to work with, so we extract and decode the 

314 names from cursor.description and index those as the primary 

315 result row target keys. 

316 

317 The remaining fairly common case is that of the textual SQL 

318 that includes at least partial column information; this is when 

319 we use a :class:`_expression.TextualSelect` construct. 

320 This construct may have 

321 unordered or ordered column information. In the ordered case, we 

322 merge the cursor.description and the compiled construct's information 

323 positionally, and warn if there are additional description names 

324 present, however we still decode the names in cursor.description 

325 as we don't have a guarantee that the names in the columns match 

326 on these. In the unordered case, we match names in cursor.description 

327 to that of the compiled construct based on name matching. 

328 In both of these cases, the cursor.description names and the column 

329 expression objects and names are indexed as result row target keys. 

330 

331 The final case is much less common, where we have a compiled 

332 non-textual SQL expression construct, but the number of columns 

333 in cursor.description doesn't match what's in the compiled 

334 construct. We make the guess here that there might be textual 

335 column expressions in the compiled construct that themselves include 

336 a comma in them causing them to split. We do the same name-matching 

337 as with textual non-ordered columns. 

338 

339 The name-matched system of merging is the same as that used by 

340 SQLAlchemy for all cases up through te 0.9 series. Positional 

341 matching for compiled SQL expressions was introduced in 1.0 as a 

342 major performance feature, and positional matching for textual 

343 :class:`_expression.TextualSelect` objects in 1.1. 

344 As name matching is no longer 

345 a common case, it was acceptable to factor it into smaller generator- 

346 oriented methods that are easier to understand, but incur slightly 

347 more performance overhead. 

348 

349 """ 

350 

351 case_sensitive = context.dialect.case_sensitive 

352 

353 if ( 

354 num_ctx_cols 

355 and cols_are_ordered 

356 and not textual_ordered 

357 and num_ctx_cols == len(cursor_description) 

358 ): 

359 self._keys = [elem[0] for elem in result_columns] 

360 # pure positional 1-1 case; doesn't need to read 

361 # the names from cursor.description 

362 

363 # this metadata is safe to cache because we are guaranteed 

364 # to have the columns in the same order for new executions 

365 self._safe_for_cache = True 

366 return [ 

367 ( 

368 idx, 

369 idx, 

370 rmap_entry[RM_OBJECTS], 

371 rmap_entry[RM_NAME].lower() 

372 if not case_sensitive 

373 else rmap_entry[RM_NAME], 

374 rmap_entry[RM_RENDERED_NAME], 

375 context.get_result_processor( 

376 rmap_entry[RM_TYPE], 

377 rmap_entry[RM_RENDERED_NAME], 

378 cursor_description[idx][1], 

379 ), 

380 None, 

381 ) 

382 for idx, rmap_entry in enumerate(result_columns) 

383 ] 

384 else: 

385 

386 # name-based or text-positional cases, where we need 

387 # to read cursor.description names 

388 

389 if textual_ordered: 

390 self._safe_for_cache = True 

391 # textual positional case 

392 raw_iterator = self._merge_textual_cols_by_position( 

393 context, cursor_description, result_columns 

394 ) 

395 elif num_ctx_cols: 

396 # compiled SQL with a mismatch of description cols 

397 # vs. compiled cols, or textual w/ unordered columns 

398 # the order of columns can change if the query is 

399 # against a "select *", so not safe to cache 

400 self._safe_for_cache = False 

401 raw_iterator = self._merge_cols_by_name( 

402 context, 

403 cursor_description, 

404 result_columns, 

405 loose_column_name_matching, 

406 ) 

407 else: 

408 # no compiled SQL, just a raw string, order of columns 

409 # can change for "select *" 

410 self._safe_for_cache = False 

411 raw_iterator = self._merge_cols_by_none( 

412 context, cursor_description 

413 ) 

414 

415 return [ 

416 ( 

417 idx, 

418 ridx, 

419 obj, 

420 cursor_colname, 

421 cursor_colname, 

422 context.get_result_processor( 

423 mapped_type, cursor_colname, coltype 

424 ), 

425 untranslated, 

426 ) 

427 for ( 

428 idx, 

429 ridx, 

430 cursor_colname, 

431 mapped_type, 

432 coltype, 

433 obj, 

434 untranslated, 

435 ) in raw_iterator 

436 ]

438 def _colnames_from_description(self, context, cursor_description): 

439 """Extract column names and data types from a cursor.description. 

440 

441 Applies unicode decoding, column translation, "normalization", 

442 and case sensitivity rules to the names based on the dialect. 

443 

444 """ 

445 

446 dialect = context.dialect 

447 case_sensitive = dialect.case_sensitive 

448 translate_colname = context._translate_colname 

449 description_decoder = ( 

450 dialect._description_decoder 

451 if dialect.description_encoding 

452 else None 

453 ) 

454 normalize_name = ( 

455 dialect.normalize_name if dialect.requires_name_normalize else None 

456 ) 

457 untranslated = None 

458 

459 self._keys = [] 

460 

461 for idx, rec in enumerate(cursor_description): 

462 colname = rec[0] 

463 coltype = rec[1] 

464 

465 if description_decoder: 

466 colname = description_decoder(colname) 

467 

468 if translate_colname: 

469 colname, untranslated = translate_colname(colname) 

470 

471 if normalize_name: 

472 colname = normalize_name(colname) 

473 

474 self._keys.append(colname) 

475 if not case_sensitive: 

476 colname = colname.lower() 

477 

478 yield idx, colname, untranslated, coltype

480 def _merge_textual_cols_by_position( 

481 self, context, cursor_description, result_columns 

482 ): 

483 num_ctx_cols = len(result_columns) if result_columns else None 

484 

485 if num_ctx_cols > len(cursor_description): 

486 util.warn( 

487 "Number of columns in textual SQL (%d) is " 

488 "smaller than number of columns requested (%d)" 

489 % (num_ctx_cols, len(cursor_description)) 

490 ) 

491 seen = set() 

492 for ( 

493 idx, 

494 colname, 

495 untranslated, 

496 coltype, 

497 ) in self._colnames_from_description(context, cursor_description): 

498 if idx < num_ctx_cols: 

499 ctx_rec = result_columns[idx] 

500 obj = ctx_rec[RM_OBJECTS] 

501 ridx = idx 

502 mapped_type = ctx_rec[RM_TYPE] 

503 if obj[0] in seen: 

504 raise exc.InvalidRequestError( 

505 "Duplicate column expression requested " 

506 "in textual SQL: %r" % obj[0] 

507 ) 

508 seen.add(obj[0]) 

509 else: 

510 mapped_type = sqltypes.NULLTYPE 

511 obj = None 

512 ridx = None 

513 yield idx, ridx, colname, mapped_type, coltype, obj, untranslated

553 @classmethod 

554 def _create_description_match_map( 

555 cls, 

556 result_columns, 

557 case_sensitive=True, 

558 loose_column_name_matching=False, 

559 ): 

560 """when matching cursor.description to a set of names that are present 

561 in a Compiled object, as is the case with TextualSelect, get all the 

562 names we expect might match those in cursor.description. 

563 """ 

564 

565 d = {} 

566 for ridx, elem in enumerate(result_columns): 

567 key = elem[RM_RENDERED_NAME] 

568 

569 if not case_sensitive: 

570 key = key.lower() 

571 if key in d: 

572 # conflicting keyname - just add the column-linked objects 

573 # to the existing record. if there is a duplicate column 

574 # name in the cursor description, this will allow all of those 

575 # objects to raise an ambiguous column error 

576 e_name, e_obj, e_type, e_ridx = d[key] 

577 d[key] = e_name, e_obj + elem[RM_OBJECTS], e_type, ridx 

578 else: 

579 d[key] = (elem[RM_NAME], elem[RM_OBJECTS], elem[RM_TYPE], ridx) 

580 

581 if loose_column_name_matching: 

582 # when using a textual statement with an unordered set 

583 # of columns that line up, we are expecting the user 

584 # to be using label names in the SQL that match to the column 

585 # expressions. Enable more liberal matching for this case; 

586 # duplicate keys that are ambiguous will be fixed later. 

587 for r_key in elem[RM_OBJECTS]: 

588 d.setdefault( 

589 r_key, 

590 (elem[RM_NAME], elem[RM_OBJECTS], elem[RM_TYPE], ridx), 

591 ) 

592 

593 return d

981class BufferedRowCursorFetchStrategy(CursorFetchStrategy): 

982 """A cursor fetch strategy with row buffering behavior. 

983 

984 This strategy buffers the contents of a selection of rows 

985 before ``fetchone()`` is called. This is to allow the results of 

986 ``cursor.description`` to be available immediately, when 

987 interfacing with a DB-API that requires rows to be consumed before 

988 this information is available (currently psycopg2, when used with 

989 server-side cursors). 

990 

991 The pre-fetching behavior fetches only one row initially, and then 

992 grows its buffer size by a fixed amount with each successive need 

993 for additional rows up the ``max_row_buffer`` size, which defaults 

994 to 1000:: 

995

997 

998 result = conn.execution_options( 

999 stream_results=True, max_row_buffer=50 

1000 ).execute(text("select * from table"))

1002 .. versionadded:: 1.4 ``max_row_buffer`` may now exceed 1000 rows. 

1003 

1004 .. seealso:: 

1005 

1006 :ref:`psycopg2_execution_options`

1012 self,

1015 growth_factor=5, 

1016 initial_buffer=None,

1043 new_rows = dbapi_cursor.fetchall() 

1044 else: 

1045 new_rows = dbapi_cursor.fetchmany(size)

1065 )

1070 result, dbapi_cursor 

1071 ) 

1072

1074 if not self._rowbuffer: 

1075 self._buffer_rows(result, dbapi_cursor) 

1076 if not self._rowbuffer: 

1077 try: 

1078 result._soft_close(hard=hard_close) 

1079 except BaseException as e: 

1080 self.handle_exception(result, dbapi_cursor, e) 

1081 return None 

1082 return self._rowbuffer.popleft()

1089 lb = len(buf)

1093 except BaseException as e: 

1094 self.handle_exception(result, dbapi_cursor, e) 

1095

1097 self._rowbuffer = collections.deque(buf[size:]) 

1098 return result

1102 ret = list(self._rowbuffer) + list(dbapi_cursor.fetchall()) 

1103 self._rowbuffer.clear() 

1104 result._soft_close() 

1105 return ret 

1106 except BaseException as e: 

1107 self.handle_exception(result, dbapi_cursor, e)

39def instances(cursor, context): 

40 """Return a :class:`.Result` given an ORM query context. 

41 

42 :param cursor: a :class:`.CursorResult`, generated by a statement 

43 which came from :class:`.ORMCompileState` 

44 

45 :param context: a :class:`.QueryContext` object 

46 

47 :return: a :class:`.Result` object representing ORM results 

48 

49 .. versionchanged:: 1.4 The instances() function now uses 

50 :class:`.Result` objects and has an all new interface. 

51 

52 """ 

53 

54 context.runid = _new_runid() 

55 context.post_load_paths = {} 

56 

57 compile_state = context.compile_state 

58 filtered = compile_state._has_mapper_entities 

59 single_entity = ( 

60 not context.load_options._only_return_tuples 

61 and len(compile_state._entities) == 1 

62 and compile_state._entities[0].supports_single_entity 

63 ) 

64 

65 try: 

66 (process, labels, extra) = list( 

67 zip( 

68 *[ 

69 query_entity.row_processor(context, cursor) 

70 for query_entity in context.compile_state._entities 

71 ] 

72 ) 

73 ) 

74 

75 if context.yield_per and ( 

76 context.loaders_require_buffering 

77 or context.loaders_require_uniquing 

78 ): 

79 raise sa_exc.InvalidRequestError( 

80 "Can't use yield_per with eager loaders that require uniquing " 

81 "or row buffering, e.g. joinedload() against collections " 

82 "or subqueryload(). Consider the selectinload() strategy " 

83 "for better flexibility in loading objects." 

84 ) 

85 

86 except Exception: 

87 with util.safe_reraise(): 

88 cursor.close() 

89 

90 def _no_unique(entry): 

91 raise sa_exc.InvalidRequestError( 

92 "Can't use the ORM yield_per feature in conjunction with unique()" 

93 ) 

94 

95 row_metadata = SimpleResultMetaData( 

96 labels, 

97 extra, 

98 _unique_filters=[ 

99 _no_unique 

100 if context.yield_per 

101 else id 

102 if ent.use_id_for_hash 

103 else None 

104 for ent in context.compile_state._entities 

105 ], 

106 ) 

107 

108 def chunks(size): 

109 while True: 

110 yield_per = size 

111 

112 context.partials = {} 

113 

114 if yield_per: 

115 fetch = cursor.fetchmany(yield_per) 

116 

117 if not fetch: 

118 break 

119 else: 

120 fetch = cursor._raw_all_rows() 

121 

122 if single_entity: 

123 proc = process[0] 

124 rows = [proc(row) for row in fetch] 

125 else: 

126 rows = [ 

127 tuple([proc(row) for proc in process]) for row in fetch 

128 ] 

129 

130 for path, post_load in context.post_load_paths.items(): 

131 post_load.invoke(context, path) 

132 

133 yield rows 

134 

135 if not yield_per: 

136 break 

137 

138 if context.execution_options.get("prebuffer_rows", False): 

139 # this is a bit of a hack at the moment. 

140 # I would rather have some option in the result to pre-buffer 

141 # internally. 

142 _prebuffered = list(chunks(None)) 

143 

144 def chunks(size): 

145 return iter(_prebuffered) 

146 

147 result = ChunkedIteratorResult( 

148 row_metadata, 

149 chunks, 

150 source_supports_scalars=single_entity, 

151 raw=cursor, 

152 dynamic_yield_per=cursor.context._is_server_side, 

153 ) 

154 

155 # filtered and single_entity are used to indicate to legacy Query that the 

156 # query has ORM entities, so legacy deduping and scalars should be called 

157 # on the result. 

158 result._attributes = result._attributes.union( 

159 dict(filtered=filtered, is_single_entity=single_entity) 

160 ) 

161 

162 # multi_row_eager_loaders OTOH is specific to joinedload. 

163 if context.compile_state.multi_row_eager_loaders: 

164 

165 def require_unique(obj): 

166 raise sa_exc.InvalidRequestError( 

167 "The unique() method must be invoked on this Result, " 

168 "as it contains results that include joined eager loads " 

169 "against collections" 

170 ) 

171 

172 result._unique_filter_state = (None, require_unique) 

173 

174 if context.yield_per: 

175 result.yield_per(context.yield_per) 

176 

177 return result

181def merge_frozen_result(session, statement, frozen_result, load=True): 

182 """Merge a :class:`_engine.FrozenResult` back into a :class:`_orm.Session`, 

183 returning a new :class:`_engine.Result` object with :term:`persistent` 

184 objects. 

185 

186 See the section :ref:`do_orm_execute_re_executing` for an example. 

187 

188 .. seealso:: 

189 

190 :ref:`do_orm_execute_re_executing` 

191 

192 :meth:`_engine.Result.freeze` 

193 

194 :class:`_engine.FrozenResult` 

195 

196 """ 

197 querycontext = util.preloaded.orm_context 

198 

199 if load: 

200 # flush current contents if we expect to load data 

201 session._autoflush() 

202 

203 ctx = querycontext.ORMSelectCompileState._create_entities_collection( 

204 statement, legacy=False 

205 ) 

206 

207 autoflush = session.autoflush 

208 try: 

209 session.autoflush = False 

210 mapped_entities = [ 

211 i 

212 for i, e in enumerate(ctx._entities) 

213 if isinstance(e, querycontext._MapperEntity) 

214 ] 

215 keys = [ent._label_name for ent in ctx._entities] 

216 

217 keyed_tuple = result_tuple( 

218 keys, [ent._extra_entities for ent in ctx._entities] 

219 ) 

220 

221 result = [] 

222 for newrow in frozen_result.rewrite_rows(): 

223 for i in mapped_entities: 

224 if newrow[i] is not None: 

225 newrow[i] = session._merge( 

226 attributes.instance_state(newrow[i]), 

227 attributes.instance_dict(newrow[i]), 

228 load=load, 

229 _recursive={}, 

230 _resolve_conflict_map={}, 

231 ) 

232 

233 result.append(keyed_tuple(newrow)) 

234 

235 return frozen_result.with_new_rows(result) 

236 finally: 

237 session.autoflush = autoflush

246def merge_result(query, iterator, load=True): 

247 """Merge a result into this :class:`.Query` object's Session.""" 

248 

249 querycontext = util.preloaded.orm_context 

250 

251 session = query.session 

252 if load: 

253 # flush current contents if we expect to load data 

254 session._autoflush() 

255 

256 # TODO: need test coverage and documentation for the FrozenResult 

257 # use case. 

258 if isinstance(iterator, FrozenResult): 

259 frozen_result = iterator 

260 iterator = iter(frozen_result.data) 

261 else: 

262 frozen_result = None 

263 

264 ctx = querycontext.ORMSelectCompileState._create_entities_collection( 

265 query, legacy=True 

266 ) 

267 

268 autoflush = session.autoflush 

269 try: 

270 session.autoflush = False 

271 single_entity = not frozen_result and len(ctx._entities) == 1 

272 

273 if single_entity: 

274 if isinstance(ctx._entities[0], querycontext._MapperEntity): 

275 result = [ 

276 session._merge( 

277 attributes.instance_state(instance), 

278 attributes.instance_dict(instance), 

279 load=load, 

280 _recursive={}, 

281 _resolve_conflict_map={}, 

282 ) 

283 for instance in iterator 

284 ] 

285 else: 

286 result = list(iterator) 

287 else: 

288 mapped_entities = [ 

289 i 

290 for i, e in enumerate(ctx._entities) 

291 if isinstance(e, querycontext._MapperEntity) 

292 ] 

293 result = [] 

294 keys = [ent._label_name for ent in ctx._entities] 

295 

296 keyed_tuple = result_tuple( 

297 keys, [ent._extra_entities for ent in ctx._entities] 

298 ) 

299 

300 for row in iterator: 

301 newrow = list(row) 

302 for i in mapped_entities: 

303 if newrow[i] is not None: 

304 newrow[i] = session._merge( 

305 attributes.instance_state(newrow[i]), 

306 attributes.instance_dict(newrow[i]), 

307 load=load, 

308 _recursive={}, 

309 _resolve_conflict_map={}, 

310 ) 

311 result.append(keyed_tuple(newrow)) 

312 

313 if frozen_result: 

314 return frozen_result.with_data(result) 

315 else: 

316 return iter(result) 

317 finally: 

318 session.autoflush = autoflush

321def get_from_identity(session, mapper, key, passive): 

322 """Look up the given key in the given session's identity map, 

323 check the object for expired state if found. 

324 

325 """ 

326 instance = session.identity_map.get(key) 

327 if instance is not None: 

328 

329 state = attributes.instance_state(instance) 

330 

331 if mapper.inherits and not state.mapper.isa(mapper): 

332 return attributes.PASSIVE_CLASS_MISMATCH 

333 

334 # expired - ensure it still exists 

335 if state.expired: 

336 if not passive & attributes.SQL_OK: 

337 # TODO: no coverage here 

338 return attributes.PASSIVE_NO_RESULT 

339 elif not passive & attributes.RELATED_OBJECT_OK: 

340 # this mode is used within a flush and the instance's 

341 # expired state will be checked soon enough, if necessary. 

342 # also used by immediateloader for a mutually-dependent 

343 # o2m->m2m load, :ticket:`6301` 

344 return instance 

345 try: 

346 state._load_expired(state, passive) 

347 except orm_exc.ObjectDeletedError: 

348 session._remove_newly_deleted([state]) 

349 return None 

350 return instance 

351 else: 

352 return None

389def load_on_pk_identity( 

390 session, 

391 statement, 

392 primary_key_identity, 

393 load_options=None, 

394 refresh_state=None, 

395 with_for_update=None, 

396 only_load_props=None, 

397 identity_token=None, 

398 no_autoflush=False, 

399 bind_arguments=util.EMPTY_DICT, 

400 execution_options=util.EMPTY_DICT, 

401): 

402 

403 """Load the given primary key identity from the database.""" 

404 

405 query = statement 

406 q = query._clone() 

407 

408 is_lambda = q._is_lambda_element 

409 

410 # TODO: fix these imports .... 

411 from .context import QueryContext, ORMCompileState 

412 

413 if load_options is None: 

414 load_options = QueryContext.default_load_options 

415 

416 compile_options = ORMCompileState.default_compile_options 

417 

418 if primary_key_identity is not None: 

419 mapper = query._propagate_attrs["plugin_subject"] 

420 

421 (_get_clause, _get_params) = mapper._get_clause 

422 

423 # None present in ident - turn those comparisons 

424 # into "IS NULL" 

425 if None in primary_key_identity: 

426 nones = set( 

427 [ 

428 _get_params[col].key 

429 for col, value in zip( 

430 mapper.primary_key, primary_key_identity 

431 ) 

432 if value is None 

433 ] 

434 ) 

435 

436 _get_clause = sql_util.adapt_criterion_to_null(_get_clause, nones) 

437 

438 if len(nones) == len(primary_key_identity): 

439 util.warn( 

440 "fully NULL primary key identity cannot load any " 

441 "object. This condition may raise an error in a future " 

442 "release." 

443 ) 

444 

445 if is_lambda: 

446 q = q.add_criteria( 

447 lambda q: q.where( 

448 sql_util._deep_annotate(_get_clause, {"_orm_adapt": True}) 

449 ), 

450 # this track_on will allow the lambda to refresh if 

451 # _get_clause goes stale due to reconfigured mapper. 

452 # however, it's not needed as the lambda otherwise tracks 

453 # on the SQL cache key of the expression. the main thing 

454 # is that the bindparam.key stays the same if the cache key 

455 # stays the same, as we are referring to the .key explicitly 

456 # in the params. 

457 # track_on=[id(_get_clause)] 

458 ) 

459 else: 

460 q._where_criteria = ( 

461 sql_util._deep_annotate(_get_clause, {"_orm_adapt": True}), 

462 ) 

463 

464 params = dict( 

465 [ 

466 (_get_params[primary_key].key, id_val) 

467 for id_val, primary_key in zip( 

468 primary_key_identity, mapper.primary_key 

469 ) 

470 ] 

471 ) 

472 else: 

473 params = None 

474 

475 if is_lambda: 

476 if with_for_update is not None or refresh_state or only_load_props: 

477 raise NotImplementedError( 

478 "refresh operation not supported with lambda statement" 

479 ) 

480 

481 version_check = False 

482 

483 _, load_options = _set_get_options( 

484 compile_options, 

485 load_options, 

486 version_check=version_check, 

487 only_load_props=only_load_props, 

488 refresh_state=refresh_state, 

489 identity_token=identity_token, 

490 ) 

491 

492 if no_autoflush: 

493 load_options += {"_autoflush": False} 

494 else: 

495 if with_for_update is not None: 

496 version_check = True 

497 q._for_update_arg = with_for_update 

498 elif query._for_update_arg is not None: 

499 version_check = True 

500 q._for_update_arg = query._for_update_arg 

501 else: 

502 version_check = False 

503 

504 if refresh_state and refresh_state.load_options: 

505 compile_options += { 

506 "_current_path": refresh_state.load_path.parent 

507 } 

508 q = q.options(*refresh_state.load_options) 

509 

510 # TODO: most of the compile_options that are not legacy only involve 

511 # this function, so try to see if handling of them can mostly be local 

512 # to here 

513 

514 q._compile_options, load_options = _set_get_options( 

515 compile_options, 

516 load_options, 

517 version_check=version_check, 

518 only_load_props=only_load_props, 

519 refresh_state=refresh_state, 

520 identity_token=identity_token, 

521 ) 

522 q._order_by = None 

523 

524 if no_autoflush: 

525 load_options += {"_autoflush": False} 

526 

527 execution_options = util.EMPTY_DICT.merge_with( 

528 execution_options, {"_sa_orm_load_options": load_options} 

529 ) 

530 result = ( 

531 session.execute( 

532 q, 

533 params=params, 

534 execution_options=execution_options, 

535 bind_arguments=bind_arguments, 

536 ) 

537 .unique() 

538 .scalars() 

539 ) 

540 

541 try: 

542 return result.one() 

543 except orm_exc.NoResultFound: 

544 return None

583 path, 

584 adapter, 

585 column_collection, 

586 with_polymorphic=None,

590):

594 with_polymorphic

598

601 path.set(compile_state.attributes, "memoized_setups", quick_populators)

623 ) 

624

626 polymorphic_discriminator is not None 

627 and polymorphic_discriminator is not mapper.polymorphic_on 

628 ): 

629 

630 if adapter: 

631 pd = adapter.columns[polymorphic_discriminator] 

632 else: 

633 pd = polymorphic_discriminator 

634 column_collection.append(pd)

649def _instance_processor( 

650 query_entity, 

651 mapper, 

652 context, 

653 result, 

654 path, 

655 adapter, 

656 only_load_props=None, 

657 refresh_state=None, 

658 polymorphic_discriminator=None, 

659 _polymorphic_from=None, 

660): 

661 """Produce a mapper level row processor callable 

662 which processes rows into mapped instances.""" 

663 

664 # note that this method, most of which exists in a closure 

665 # called _instance(), resists being broken out, as 

666 # attempts to do so tend to add significant function 

667 # call overhead. _instance() is the most 

668 # performance-critical section in the whole ORM. 

669 

670 identity_class = mapper._identity_class 

671 compile_state = context.compile_state 

672 

673 # look for "row getter" functions that have been assigned along 

674 # with the compile state that were cached from a previous load. 

675 # these are operator.itemgetter() objects that each will extract a 

676 # particular column from each row. 

677 

678 getter_key = ("getters", mapper) 

679 getters = path.get(compile_state.attributes, getter_key, None) 

680 

681 if getters is None: 

682 # no getters, so go through a list of attributes we are loading for, 

683 # and the ones that are column based will have already put information 

684 # for us in another collection "memoized_setups", which represents the 

685 # output of the LoaderStrategy.setup_query() method. We can just as 

686 # easily call LoaderStrategy.create_row_processor for each, but by 

687 # getting it all at once from setup_query we save another method call 

688 # per attribute. 

689 props = mapper._prop_set 

690 if only_load_props is not None: 

691 props = props.intersection( 

692 mapper._props[k] for k in only_load_props 

693 ) 

694 

695 quick_populators = path.get( 

696 context.attributes, "memoized_setups", _none_set 

697 ) 

698 

699 todo = [] 

700 cached_populators = { 

701 "new": [], 

702 "quick": [], 

703 "deferred": [], 

704 "expire": [], 

705 "delayed": [], 

706 "existing": [], 

707 "eager": [], 

708 } 

709 

710 if refresh_state is None: 

711 # we can also get the "primary key" tuple getter function 

712 pk_cols = mapper.primary_key 

713 

714 if adapter: 

715 pk_cols = [adapter.columns[c] for c in pk_cols] 

716 primary_key_getter = result._tuple_getter(pk_cols) 

717 else: 

718 primary_key_getter = None 

719 

720 getters = { 

721 "cached_populators": cached_populators, 

722 "todo": todo, 

723 "primary_key_getter": primary_key_getter, 

724 } 

725 for prop in props: 

726 if prop in quick_populators: 

727 # this is an inlined path just for column-based attributes. 

728 col = quick_populators[prop] 

729 if col is _DEFER_FOR_STATE: 

730 cached_populators["new"].append( 

731 (prop.key, prop._deferred_column_loader) 

732 ) 

733 elif col is _SET_DEFERRED_EXPIRED: 

734 # note that in this path, we are no longer 

735 # searching in the result to see if the column might 

736 # be present in some unexpected way. 

737 cached_populators["expire"].append((prop.key, False)) 

738 elif col is _RAISE_FOR_STATE: 

739 cached_populators["new"].append( 

740 (prop.key, prop._raise_column_loader) 

741 ) 

742 else: 

743 getter = None 

744 if adapter: 

745 # this logic had been removed for all 1.4 releases 

746 # up until 1.4.18; the adapter here is particularly 

747 # the compound eager adapter which isn't accommodated 

748 # in the quick_populators right now. The "fallback" 

749 # logic below instead took over in many more cases 

750 # until issue #6596 was identified. 

751 

752 # note there is still an issue where this codepath

1305class PostLoad(object): 

1306 """Track loaders and states for "post load" operations.""" 

1307 

1308 __slots__ = "loaders", "states", "load_keys" 

1309 

1310 def __init__(self): 

1311 self.loaders = {} 

1312 self.states = util.OrderedDict() 

1313 self.load_keys = None 

1314 

1315 def add_state(self, state, overwrite): 

1316 # the states for a polymorphic load here are all shared 

1317 # within a single PostLoad object among multiple subtypes. 

1318 # Filtering of callables on a per-subclass basis needs to be done at 

1319 # the invocation level 

1320 self.states[state] = overwrite 

1321 

1322 def invoke(self, context, path): 

1323 if not self.states: 

1324 return 

1325 path = path_registry.PathRegistry.coerce(path) 

1326 for token, limit_to_mapper, loader, arg, kw in self.loaders.values(): 

1327 states = [ 

1328 (state, overwrite) 

1329 for state, overwrite in self.states.items() 

1330 if state.manager.mapper.isa(limit_to_mapper) 

1331 ] 

1332 if states: 

1333 loader(context, path, states, self.load_keys, *arg, **kw) 

1334 self.states.clear() 

1335 

1336 @classmethod 

1337 def for_context(cls, context, path, only_load_props): 

1338 pl = context.post_load_paths.get(path.path) 

1339 if pl is not None and only_load_props: 

1340 pl.load_keys = only_load_props 

1341 return pl 

1342 

1343 @classmethod 

1344 def path_exists(self, context, path, key): 

1345 return ( 

1346 path.path in context.post_load_paths 

1347 and key in context.post_load_paths[path.path].loaders 

1348 ) 

1349 

1350 @classmethod 

1351 def callable_for_path( 

1352 cls, context, path, limit_to_mapper, token, loader_callable, *arg, **kw 

1353 ): 

1354 if path.path in context.post_load_paths: 

1355 pl = context.post_load_paths[path.path] 

1356 else: 

1357 pl = context.post_load_paths[path.path] = PostLoad() 

1358 pl.loaders[token] = (token, limit_to_mapper, loader_callable, arg, kw)

1363 

1364 # assert mapper is _state_mapper(state) 

1365 session = state.session 

1366 if not session:

1373 

1374 result = False 

1375 

1376 no_autoflush = ( 

1377 bool(passive & attributes.NO_AUTOFLUSH) or state.session.autocommit

1380 # in the case of inheritance, particularly concrete and abstract 

1381 # concrete inheritance, the class manager might have some keys

1399 # from_statement() allows this.

1401 # with adapt_on_names set up. the other option is to make the 

1402 # aliased() against a subquery which affects the SQL. 

1403

1407 strategy_options.Load(mapper).undefer("*")

1412 stmt, 

1413 None,

1417 )

1419 if result is False: 

1420 if has_key: 

1421 identity_key = state.key 

1422 else: 

1423 # this codepath is rare - only valid when inside a flush, and the 

1424 # object is becoming persistent but hasn't yet been assigned 

1425 # an identity_key.

1427 pk_attrs = [ 

1428 mapper._columntoproperty[col].key for col in mapper.primary_key 

1429 ]

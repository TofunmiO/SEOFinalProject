55 def _make_extra_froms(self, statement): 

56 froms = [] 

57 

58 all_tables = list(sql_util.tables_from_leftmost(statement.table)) 

59 seen = {all_tables[0]} 

60 

61 for crit in statement._where_criteria: 

62 for item in _from_objects(crit): 

63 if not seen.intersection(item._cloned_set): 

64 froms.append(item) 

65 seen.update(item._cloned_set) 

66 

67 froms.extend(all_tables[1:]) 

68 return froms

70 def _process_multi_values(self, statement): 

71 if not statement._supports_multi_parameters: 

72 raise exc.InvalidRequestError( 

73 "%s construct does not support " 

74 "multiple parameter sets." % statement.__visit_name__.upper() 

75 ) 

76 

77 for parameters in statement._multi_values: 

78 multi_parameters = [ 

79 { 

80 c.key: value 

81 for c, value in zip(statement.table.c, parameter_set) 

82 } 

83 if isinstance(parameter_set, collections_abc.Sequence) 

84 else parameter_set 

85 for parameter_set in parameters 

86 ] 

87 

88 if self._no_parameters: 

89 self._no_parameters = False 

90 self._has_multi_parameters = True 

91 self._multi_parameters = multi_parameters 

92 self._dict_parameters = self._multi_parameters[0] 

93 elif not self._has_multi_parameters: 

94 self._cant_mix_formats_error()

106 def _process_ordered_values(self, statement):

511 def values(self, *args, **kwargs): 

512 r"""Specify a fixed VALUES clause for an INSERT statement, or the SET 

513 clause for an UPDATE. 

514 

515 Note that the :class:`_expression.Insert` and 

516 :class:`_expression.Update` 

517 constructs support 

518 per-execution time formatting of the VALUES and/or SET clauses, 

519 based on the arguments passed to :meth:`_engine.Connection.execute`. 

520 However, the :meth:`.ValuesBase.values` method can be used to "fix" a 

521 particular set of parameters into the statement. 

522 

523 Multiple calls to :meth:`.ValuesBase.values` will produce a new 

524 construct, each one with the parameter list modified to include 

525 the new parameters sent. In the typical case of a single 

526 dictionary of parameters, the newly passed keys will replace 

527 the same keys in the previous construct. In the case of a list-based 

528 "multiple values" construct, each new list of values is extended 

529 onto the existing list of values. 

530 

531 :param \**kwargs: key value pairs representing the string key 

532 of a :class:`_schema.Column` 

533 mapped to the value to be rendered into the 

534 VALUES or SET clause:: 

535 

536 users.insert().values(name="some name") 

537 

538 users.update().where(users.c.id==5).values(name="some name") 

539 

540 :param \*args: As an alternative to passing key/value parameters, 

541 a dictionary, tuple, or list of dictionaries or tuples can be passed 

542 as a single positional argument in order to form the VALUES or 

543 SET clause of the statement. The forms that are accepted vary 

544 based on whether this is an :class:`_expression.Insert` or an 

545 :class:`_expression.Update` construct. 

546 

547 For either an :class:`_expression.Insert` or 

548 :class:`_expression.Update` 

549 construct, a single dictionary can be passed, which works the same as 

550 that of the kwargs form:: 

551 

552 users.insert().values({"name": "some name"}) 

553 

554 users.update().values({"name": "some new name"}) 

555 

556 Also for either form but more typically for the 

557 :class:`_expression.Insert` construct, a tuple that contains an 

558 entry for every column in the table is also accepted:: 

559 

560 users.insert().values((5, "some name")) 

561 

562 The :class:`_expression.Insert` construct also supports being 

563 passed a list of dictionaries or full-table-tuples, which on the 

564 server will render the less common SQL syntax of "multiple values" - 

565 this syntax is supported on backends such as SQLite, PostgreSQL, 

566 MySQL, but not necessarily others:: 

567 

568 users.insert().values([ 

569 {"name": "some name"}, 

570 {"name": "some other name"}, 

571 {"name": "yet another name"}, 

572 ]) 

573 

574 The above form would render a multiple VALUES statement similar to:: 

575 

576 INSERT INTO users (name) VALUES 

577 (:name_1), 

578 (:name_2), 

579 (:name_3) 

580 

581 It is essential to note that **passing multiple values is 

582 NOT the same as using traditional executemany() form**. The above 

583 syntax is a **special** syntax not typically used. To emit an 

584 INSERT statement against multiple rows, the normal method is 

585 to pass a multiple values list to the 

586 :meth:`_engine.Connection.execute` 

587 method, which is supported by all database backends and is generally 

588 more efficient for a very large number of parameters. 

589 

590 .. seealso:: 

591 

592 :ref:`execute_multiple` - an introduction to 

593 the traditional Core method of multiple parameter set 

594 invocation for INSERTs and other statements. 

595 

596 .. versionchanged:: 1.0.0 an INSERT that uses a multiple-VALUES 

597 clause, even a list of length one, 

598 implies that the :paramref:`_expression.Insert.inline` 

599 flag is set to 

600 True, indicating that the statement will not attempt to fetch 

601 the "last inserted primary key" or other defaults. The 

602 statement deals with an arbitrary number of rows, so the 

603 :attr:`_engine.CursorResult.inserted_primary_key` 

604 accessor does not 

605 apply. 

606 

607 .. versionchanged:: 1.0.0 A multiple-VALUES INSERT now supports 

608 columns with Python side default values and callables in the 

609 same way as that of an "executemany" style of invocation; the 

610 callable is invoked for each row. See :ref:`bug_3288` 

611 for other details. 

612 

613 The UPDATE construct also supports rendering the SET parameters 

614 in a specific order. For this feature refer to the 

615 :meth:`_expression.Update.ordered_values` method. 

616 

617 .. seealso:: 

618 

619 :meth:`_expression.Update.ordered_values` 

620 

621 

622 """ 

623 if args: 

624 # positional case. this is currently expensive. we don't 

625 # yet have positional-only args so we have to check the length. 

626 # then we need to check multiparams vs. single dictionary. 

627 # since the parameter format is needed in order to determine 

628 # a cache key, we need to determine this up front. 

629 arg = args[0] 

630 

631 if kwargs: 

632 raise exc.ArgumentError( 

633 "Can't pass positional and kwargs to values() " 

634 "simultaneously" 

635 ) 

636 elif len(args) > 1: 

637 raise exc.ArgumentError( 

638 "Only a single dictionary/tuple or list of " 

639 "dictionaries/tuples is accepted positionally." 

640 ) 

641 

642 elif not self._preserve_parameter_order and isinstance( 

643 arg, collections_abc.Sequence 

644 ): 

645 

646 if arg and isinstance(arg[0], (list, dict, tuple)): 

647 self._multi_values += (arg,) 

648 return 

649 

650 # tuple values 

651 arg = {c.key: value for c, value in zip(self.table.c, arg)} 

652 elif self._preserve_parameter_order and not isinstance( 

653 arg, collections_abc.Sequence 

654 ): 

655 raise ValueError( 

656 "When preserve_parameter_order is True, " 

657 "values() only accepts a list of 2-tuples" 

658 ) 

659 

660 else: 

661 # kwarg path. this is the most common path for non-multi-params 

662 # so this is fairly quick. 

663 arg = kwargs 

664 if args: 

665 raise exc.ArgumentError( 

666 "Only a single dictionary/tuple or list of " 

667 "dictionaries/tuples is accepted positionally." 

668 ) 

669 

670 # for top level values(), convert literals to anonymous bound 

671 # parameters at statement construction time, so that these values can 

672 # participate in the cache key process like any other ClauseElement. 

673 # crud.py now intercepts bound parameters with unique=True from here 

674 # and ensures they get the "crud"-style name when rendered. 

675 

676 if self._preserve_parameter_order: 

677 arg = [ 

678 ( 

679 coercions.expect(roles.DMLColumnRole, k), 

680 coercions.expect( 

681 roles.ExpressionElementRole, 

682 v, 

683 type_=NullType(), 

684 is_crud=True, 

685 ), 

686 ) 

687 for k, v in arg 

688 ] 

689 self._ordered_values = arg 

690 else: 

691 arg = { 

692 coercions.expect(roles.DMLColumnRole, k): coercions.expect( 

693 roles.ExpressionElementRole, 

694 v, 

695 type_=NullType(), 

696 is_crud=True, 

697 ) 

698 for k, v in arg.items() 

699 } 

700 if self._values: 

701 self._values = self._values.union(arg) 

702 else: 

703 self._values = util.immutabledict(arg)

706 @_exclusive_against(

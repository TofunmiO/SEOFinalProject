8"""Visitor/traversal interface and library functions. 

9 

10SQLAlchemy schema and expression constructs rely on a Python-centric 

11version of the classic "visitor" pattern as the primary way in which 

12they apply functionality. The most common use of this pattern 

13is statement compilation, where individual expression classes match 

14up to rendering methods that produce a string result. Beyond this, 

15the visitor system is also used to inspect expressions for various 

16information and patterns, as well as for the purposes of applying 

17transformations to expressions. 

18 

19Examples of how the visit system is used can be seen in the source code 

20of for example the ``sqlalchemy.sql.util`` and the ``sqlalchemy.sql.compiler`` 

21modules. Some background on clause adaption is also at 

22http://techspot.zzzeek.org/2008/01/23/expression-transformations/ . 

23 

24"""

48def _generate_compiler_dispatch(cls): 

49 """Generate a _compiler_dispatch() external traversal on classes with a 

50 __visit_name__ attribute. 

51 

52 """ 

53 visit_name = cls.__visit_name__ 

54 

55 if "_compiler_dispatch" in cls.__dict__: 

56 # class has a fixed _compiler_dispatch() method. 

57 # copy it to "original" so that we can get it back if 

58 # sqlalchemy.ext.compiles overrides it. 

59 cls._original_compiler_dispatch = cls._compiler_dispatch 

60 return 

61 

62 if not isinstance(visit_name, util.compat.string_types): 

63 raise exc.InvalidRequestError( 

64 "__visit_name__ on class %s must be a string at the class level" 

65 % cls.__name__ 

66 ) 

67 

68 name = "visit_%s" % visit_name 

69 getter = operator.attrgetter(name) 

70 

71 def _compiler_dispatch(self, visitor, **kw): 

72 """Look for an attribute named "visit_<visit_name>" on the 

73 visitor, and call it with the same kw params. 

74 

75 """ 

76 try: 

77 meth = getter(visitor) 

78 except AttributeError as err: 

79 return visitor.visit_unsupported_compilation(self, err, **kw) 

80 

81 else: 

82 return meth(self, **kw) 

83 

84 cls._compiler_dispatch = ( 

85 cls._original_compiler_dispatch 

86 ) = _compiler_dispatch

89class TraversibleType(type): 

90 """Metaclass which assigns dispatch attributes to various kinds of 

91 "visitable" classes. 

92 

93 Attributes include: 

94 

95 * The ``_compiler_dispatch`` method, corresponding to ``__visit_name__``. 

96 This is called "external traversal" because the caller of each visit() 

97 method is responsible for sub-traversing the inner elements of each

102 ``_cache_key_traversal``, ``_copy_internals_traversal``, generated from 

103 an optional ``_traverse_internals`` collection of symbols which comes 

104 from the :class:`.InternalTraversal` list of symbols. This is called 

105 "internal traversal" MARKMARK 

106 

107 """

156class _InternalTraversalType(type): 

157 def __init__(cls, clsname, bases, clsdict): 

158 if cls.__name__ in ("InternalTraversal", "ExtendedInternalTraversal"): 

159 lookup = {} 

160 for key, sym in clsdict.items(): 

161 if key.startswith("dp_"): 

162 visit_key = key.replace("dp_", "visit_") 

163 sym_name = sym.name 

164 assert sym_name not in lookup, sym_name 

165 lookup[sym] = lookup[sym_name] = visit_key 

166 if hasattr(cls, "_dispatch_lookup"): 

167 lookup.update(cls._dispatch_lookup) 

168 cls._dispatch_lookup = lookup 

169 

170 super(_InternalTraversalType, cls).__init__(clsname, bases, clsdict)

173def _generate_dispatcher(visitor, internal_dispatch, method_name): 

174 names = [] 

175 for attrname, visit_sym in internal_dispatch: 

176 meth = visitor.dispatch(visit_sym) 

177 if meth: 

178 visit_name = ExtendedInternalTraversal._dispatch_lookup[visit_sym] 

179 names.append((attrname, visit_name)) 

180 

181 code = ( 

182 (" return [\n") 

183 + ( 

184 ", \n".join( 

185 " (%r, self.%s, visitor.%s)" 

186 % (attrname, attrname, visit_name) 

187 for attrname, visit_name in names 

188 ) 

189 ) 

190 + ("\n ]\n") 

191 ) 

192 meth_text = ("def %s(self, visitor):\n" % method_name) + code + "\n" 

193 # print(meth_text) 

194 return langhelpers._exec_code_in_env(meth_text, {}, method_name)

197class InternalTraversal(util.with_metaclass(_InternalTraversalType, object)): 

198 r"""Defines visitor symbols used for internal traversal. 

199 

200 The :class:`.InternalTraversal` class is used in two ways. One is that 

201 it can serve as the superclass for an object that implements the 

202 various visit methods of the class. The other is that the symbols 

203 themselves of :class:`.InternalTraversal` are used within 

204 the ``_traverse_internals`` collection. Such as, the :class:`.Case` 

205 object defines ``_traverse_internals`` as :: 

206 

207 _traverse_internals = [ 

208 ("value", InternalTraversal.dp_clauseelement), 

209 ("whens", InternalTraversal.dp_clauseelement_tuples), 

210 ("else_", InternalTraversal.dp_clauseelement), 

211 ] 

212 

213 Above, the :class:`.Case` class indicates its internal state as the 

214 attributes named ``value``, ``whens``, and ``else_``. They each 

215 link to an :class:`.InternalTraversal` method which indicates the type 

216 of datastructure referred towards. 

217 

218 Using the ``_traverse_internals`` structure, objects of type 

219 :class:`.InternalTraversible` will have the following methods automatically 

220 implemented: 

221 

222 * :meth:`.Traversible.get_children` 

223 

224 * :meth:`.Traversible._copy_internals` 

225 

226 * :meth:`.Traversible._gen_cache_key` 

227 

228 Subclasses can also implement these methods directly, particularly for the 

229 :meth:`.Traversible._copy_internals` method, when special steps 

230 are needed. 

231 

232 .. versionadded:: 1.4 

233 

234 """ 

235 

236 def dispatch(self, visit_symbol): 

237 """Given a method from :class:`.InternalTraversal`, return the 

238 corresponding method on a subclass. 

239 

240 """ 

241 name = self._dispatch_lookup[visit_symbol] 

242 return getattr(self, name, None)

244 def run_generated_dispatch( 

245 self, target, internal_dispatch, generate_dispatcher_name 

246 ): 

247 try: 

248 dispatcher = target.__class__.__dict__[generate_dispatcher_name] 

249 except KeyError: 

250 # most of the dispatchers are generated up front 

251 # in sqlalchemy/sql/__init__.py -> 

252 # traversals.py-> _preconfigure_traversals(). 

253 # this block will generate any remaining dispatchers. 

254 dispatcher = self.generate_dispatch( 

255 target.__class__, internal_dispatch, generate_dispatcher_name 

256 ) 

257 return dispatcher(target, self) 

258 

259 def generate_dispatch( 

260 self, target_cls, internal_dispatch, generate_dispatcher_name 

261 ): 

262 dispatcher = _generate_dispatcher( 

263 self, internal_dispatch, generate_dispatcher_name 

264 ) 

265 # assert isinstance(target_cls, type) 

266 setattr(target_cls, generate_dispatcher_name, dispatcher) 

267 return dispatcher

626def iterate(obj, opts=util.immutabledict()): 

627 r"""Traverse the given expression structure, returning an iterator. 

628 

629 Traversal is configured to be breadth-first. 

630 

631 The central API feature used by the :func:`.visitors.iterate` 

632 function is the 

633 :meth:`_expression.ClauseElement.get_children` method of 

634 :class:`_expression.ClauseElement` objects. This method should return all 

635 the :class:`_expression.ClauseElement` objects which are associated with a 

636 particular :class:`_expression.ClauseElement` object. For example, a 

637 :class:`.Case` structure will refer to a series of 

638 :class:`_expression.ColumnElement` objects within its "whens" and "else\_" 

639 member variables. 

640 

641 :param obj: :class:`_expression.ClauseElement` structure to be traversed 

642 

643 :param opts: dictionary of iteration options. This dictionary is usually 

644 empty in modern usage. 

645 

646 """ 

647 yield obj 

648 children = obj.get_children(**opts) 

649 

650 if not children: 

651 return 

652 

653 stack = deque([children]) 

654 while stack: 

655 t_iterator = stack.popleft() 

656 for t in t_iterator: 

657 yield t 

658 stack.append(t.get_children(**opts))

693def traverse(obj, opts, visitors): 

694 """Traverse and visit the given expression structure using the default 

695 iterator. 

696 

697 e.g.:: 

698 

699 from sqlalchemy.sql import visitors 

700 

701 stmt = select(some_table).where(some_table.c.foo == 'bar')

705

707

709 which does a breadth-first traversal using a stack. 

710

712 

713 :param opts: dictionary of iteration options. This dictionary is usually 

714 empty in modern usage. 

715

717 have strings as keys, each of which would correspond to the

719 callable functions as values, each of which represents a visitor function

721

726def cloned_traverse(obj, opts, visitors): 

727 """Clone the given expression structure, allowing modifications by 

728 visitors. 

729 

730 Traversal usage is the same as that of :func:`.visitors.traverse`. 

731 The visitor functions present in the ``visitors`` dictionary may also 

732 modify the internals of the given structure as the traversal proceeds. 

733 

734 The central API feature used by the :func:`.visitors.cloned_traverse` 

735 and :func:`.visitors.replacement_traverse` functions, in addition to the 

736 :meth:`_expression.ClauseElement.get_children` 

737 function that is used to achieve 

738 the iteration, is the :meth:`_expression.ClauseElement._copy_internals` 

739 method. 

740 For a :class:`_expression.ClauseElement` 

741 structure to support cloning and replacement 

742 traversals correctly, it needs to be able to pass a cloning function into 

743 its internal members in order to make copies of them. 

744 

745 .. seealso:: 

746 

747 :func:`.visitors.traverse` 

748 

749 :func:`.visitors.replacement_traverse` 

750 

751 """ 

752 

753 cloned = {} 

754 stop_on = set(opts.get("stop_on", [])) 

755 

756 def deferred_copy_internals(obj): 

757 return cloned_traverse(obj, opts, visitors) 

758 

759 def clone(elem, **kw): 

760 if elem in stop_on: 

761 return elem 

762 else: 

763 if id(elem) not in cloned: 

764 

765 if "replace" in kw: 

766 newelem = kw["replace"](elem) 

767 if newelem is not None: 

768 cloned[id(elem)] = newelem 

769 return newelem 

770 

771 cloned[id(elem)] = newelem = elem._clone(**kw) 

772 newelem._copy_internals(clone=clone, **kw) 

773 meth = visitors.get(newelem.__visit_name__, None) 

774 if meth: 

775 meth(newelem) 

776 return cloned[id(elem)] 

777 

778 if obj is not None: 

779 obj = clone( 

780 obj, deferred_copy_internals=deferred_copy_internals, **opts 

781 ) 

782 clone = None # remove gc cycles 

783 return obj

786def replacement_traverse(obj, opts, replace): 

787 """Clone the given expression structure, allowing element 

788 replacement by a given replacement function. 

789 

790 This function is very similar to the :func:`.visitors.cloned_traverse` 

791 function, except instead of being passed a dictionary of visitors, all 

792 elements are unconditionally passed into the given replace function. 

793 The replace function then has the option to return an entirely new object 

794 which will replace the one given. If it returns ``None``, then the object 

795 is kept in place. 

796 

797 The difference in usage between :func:`.visitors.cloned_traverse` and 

798 :func:`.visitors.replacement_traverse` is that in the former case, an 

799 already-cloned object is passed to the visitor function, and the visitor 

800 function can then manipulate the internal state of the object. 

801 In the case of the latter, the visitor function should only return an 

802 entirely different object, or do nothing. 

803 

804 The use case for :func:`.visitors.replacement_traverse` is that of 

805 replacing a FROM clause inside of a SQL structure with a different one, 

806 as is a common use case within the ORM. 

807 

808 """ 

809 

810 cloned = {} 

811 stop_on = {id(x) for x in opts.get("stop_on", [])} 

812 

813 def deferred_copy_internals(obj): 

814 return replacement_traverse(obj, opts, replace) 

815 

816 def clone(elem, **kw): 

817 if ( 

818 id(elem) in stop_on 

819 or "no_replacement_traverse" in elem._annotations 

820 ): 

821 return elem 

822 else: 

823 newelem = replace(elem) 

824 if newelem is not None: 

825 stop_on.add(id(newelem)) 

826 return newelem 

827 else: 

828 # base "already seen" on id(), not hash, so that we don't 

829 # replace an Annotated element with its non-annotated one, and 

830 # vice versa 

831 id_elem = id(elem) 

832 if id_elem not in cloned: 

833 if "replace" in kw: 

834 newelem = kw["replace"](elem) 

835 if newelem is not None: 

836 cloned[id_elem] = newelem 

837 return newelem 

838 

839 cloned[id_elem] = newelem = elem._clone(**kw) 

840 newelem._copy_internals(clone=clone, **kw) 

841 return cloned[id_elem] 

842 

843 if obj is not None: 

844 obj = clone( 

845 obj, deferred_copy_internals=deferred_copy_internals, **opts 

846 ) 

847 clone = None # remove gc cycles 

848 return obj
